001package com.palamida.util.collect;
002
003import static com.google.common.base.Preconditions.checkNotNull;
004
005import java.util.Collections;
006
007/**
008 * Static utility methods which operate on or return {@link Sortedlist}s.
009 * 
010 * @author Zhenya Leonov
011 */
012final public class Sortedlists {
013
014        private Sortedlists() {
015        }
016
017        /**
018         * Creates a {@code Treelist} containing the specified initial elements
019         * sorted according to their <i>natural ordering</i>.
020         * 
021         * @param first
022         *            the first element
023         * @param rest
024         *            an array of additional elements, possibly empty
025         * @return a {@code Treelist} containing the specified initial elements
026         *         sorted according to their <i>natural ordering</i>
027         */
028        public static <E extends Comparable<? super E>> Treelist<E> newTreelist(
029                        final E first, final E... rest) {
030                checkNotNull(first);
031                checkNotNull(rest);
032                final Treelist<E> tl = Treelist.create();
033                tl.add(first);
034                Collections.addAll(tl, rest);
035                return tl;
036        }
037
038        /**
039         * Creates a {@code Skiplist} containing the specified initial elements
040         * sorted according to their <i>natural ordering</i>.
041         * 
042         * @param first
043         *            the first element
044         * @param rest
045         *            an array of additional elements, possibly empty
046         * @return a {@code Skiplist} containing the specified initial elements
047         *         sorted according to their <i>natural ordering</i>
048         */
049        public static <E extends Comparable<? super E>> Skiplist<E> newSkiplist(
050                        final E first, final E... rest) {
051                checkNotNull(first);
052                checkNotNull(rest);
053                final Skiplist<E> sl = Skiplist.create();
054                sl.add(first);
055                Collections.addAll(sl, rest);
056                return sl;
057        }
058
059}
