039/**
040 * A {@code List} optimized for efficient <a
041 * href="http://en.wikipedia.org/wiki/Random_access">random access</a> insertion
042 * and removal operations. Implements all optional list operations, and permits
043 * all elements, including {@code null}.
044 * <p>
045 * The iterators obtained from the {@link #iterator()} and
046 * {@link #listIterator()} methods are <i>fail-fast</i>. Attempts to modify the
047 * elements in this list at any time after an iterator is created, in any way
048 * except through the iterator's own remove method, will result in a
049 * {@code ConcurrentModificationException}.
050 * <p>
051 * This list is not <i>thread-safe</i>. If multiple threads modify this list
052 * concurrently it must be synchronized externally.
053 * <p>
054 * This class implements an array-based <a
055 * href="http://en.wikipedia.org/wiki/Skip_list">skip list</a> modified to
056 * provide logarithmic running time for linear list operations (e.g. insert an
057 * element at the i<i>th</i> index). Linear list operations are sometimes
058 * referred to as rank operations.
059 * <p>
060 * Invented by <a href="http://www.cs.umd.edu/~pugh/">Bill Pugh</a> in 1990, A
061 * skip list is a probabilistic data structure for maintaining items in sorted
062 * order. Strictly speaking it is impossible to make any hard guarantees
063 * regarding the worst-case performance of this class. Practical performance is
064 * <i>expected</i> to be logarithmic with an extremely high degree of
065 * probability as the list grows.
066 * <p>
067 * The following table summarizes the performance of this class compared to an
068 * {@link ArrayList} and a {@link LinkedList} (where n is the size of this list
069 * and <i>m</i> is the size of the specified collection which is iterable in
070 * linear time):
071 * <p>
072 * <table border cellpadding="3" cellspacing="1" style="width:450px;">
073 *   <tr>
074 *     <th style="text-align:center;" rowspan="2">Method</th>
075 *     <th style="text-align:center;" colspan="3">Running Time</th>
076 *   </tr>
077 *   <tr>
078 *     <td style="text-align:center;"><b>RankList</b><br>(<i>expected</i>)</td>
079 *     <td style="text-align:center;"><b>ArrayList</b><br>(<i>amortized</i>)</td>
080 *     <td style="text-align:center;"><b>LinkedList</b><br>(<i>worst-case</i>)</td>
081 *   </tr>
082 *   <tr>
083 *     <td>
084 *       {@link #addAll(Collection) addAll(Collection)}
085 *     </td>
086 *     <td style="text-align:center;" bgcolor="FFCC99"><i>O(m log n)</i></td>
087 *     <td style="text-align:center;" bgcolor="FFCCCC"><i>O(m)</i></td>
088 *     <td style="text-align:center;" bgcolor="9999CC"><i>O(m)</i></td>
089 *   </tr>
090 *   <tr>
091 *     <td>
092 *       {@link #containsAll(Collection) containsAll(Collection)}<br/>
093 *       {@link #removeAll(Collection) removeAll(Collection)}<br/>
094 *       {@link #retainAll(Collection) retainAll(Collection)}
095 *     </td>
096 *     <td style="text-align:center;" colspan="3"><i>O(m * n)</i></td>
097 *   </tr>
098 *     <td>
099 *       {@link #add(Object) add(E)}
100 *     </td>
101 *     <td style="text-align:center;" rowspan="9" bgcolor="FFCC99"><i>O(log n)</i></td>
102 *     <td style="text-align:center;" rowspan="3" bgcolor="FFCCCC"><i>O(1)</i></td>
103 *     <td style="text-align:center;" bgcolor="9999CC"><i>O(1)</i></td>
104 *   </tr>
105 *   <tr>
106 *     <td>
107 *       {@link #get(int)}<br/>
108 *       <td style="text-align:center;" rowspan="5" bgcolor="9999CC"><i>O(n)</i></td>
109 *     </td>
110 *   </tr>
111 *   <tr>
112 *     <td>
113 *       {@link #set(int, Object) set(int, E)}
114 *     </td>
115 *   </tr>
116 *   <tr>
117 *     <td>
118 *       {@link #add(int, Object) add(int, E)}<br/>
119 *     </td>
120 *     <td style="text-align:center;" rowspan="6" bgcolor="FFCCCC"><i>O(n)</i></td>
121 *     </td>
122 *   </tr>
123 *   <tr>
124 *     <td>
125 *       {@link #remove(int)}
126 *     </td>
127 *   </tr>
128 *   <tr>
129 *     <td>
130 *       {@link #listIterator(int)}
131 *     </td>
132 *   </tr>
133 *   <tr>
134 *     <td>{@link Iterator#remove()}</td>
135 *     <td style="text-align:center;" rowspan="3" bgcolor="9999CC"><i>O(1)</i></td>
136 *   </tr>
137 *   <tr>
138 *     <td>{@link ListIterator#remove()}</td>
139 *   </tr>
140 *   <tr>
141 *     <td>{@link ListIterator#add(Object) ListIterator.add(E)}</td>
142 *   </tr>
143 *   <tr>
144 *     <td>
145 *       {@link ListIterator#set(Object) ListIterator.set(E)}<br/>
146 *       {@link #clear()}
147 *     </td>
148 *     <td style="text-align:center;" colspan="3"><i>O(1)</i></td>
149 *   </tr>
150 *   <tr>
151 *     <td>
152 *       {@link #contains(Object) contains(Object)}<br/>
153 *       {@link #indexOf(Object) indexOf(Object)}<br/>
154 *       {@link #lastIndexOf(Object)}<br/>
155 *       {@link #remove(Object) remove(Object)}
156 *     </td>
157 *     <td style="text-align:center;" rowspan="4" colspan="3"><i>O(n)</i></td>
158 * </table>
159 * <p>
160 * The {@code subList} views exhibit identical time complexity, with the
161 * exception of the {@code clear()} operation which runs in linear time
162 * proportional to the size of the view.
163 * 
164 * @author Zhenya Leonov
165 * @param <E>
166 *            the type of elements maintained by this list
167 */
168public final class RankList<E> extends AbstractList<E> implements List<E>,
169                Serializable, Cloneable {
170
171        private static final long serialVersionUID = 1L;
172        private static final double P = .5;
173        private static final int MAX_LEVEL = 32;
174        private transient int size = 0;
175        private transient int level = 1;
176        private transient Random random = new Random();
177        private transient Node<E> head = new Node<E>(null, MAX_LEVEL);
178
179        private RankList() {
180                for (int i = 0; i < MAX_LEVEL; i++) {
181                        head.next[i] = head;
182                        head.dist[i] = 1;
183                }
184                head.prev = head;
185        }
186
187        /**
188         * Creates a new {@code RankList}.
189         * 
190         * @return a new {@code RankList}
191         */
192        public static <E> RankList<E> create() {
193                return new RankList<E>();
194        }

205        public static <E> RankList<E> create(final Iterable<? extends E> elements) {
206                checkNotNull(elements);
207                RankList<E> list = new RankList<E>();
208                Iterables.addAll(list, elements);
209                return list;

226        @Override
227        public Iterator<E> iterator() {
228                return listIterator();
229        }
230
231        @Override
232        public int size() {
233                return size;
234        }
235
236        @Override
237        public void add(int index, E element) {
238                checkPositionIndex(index, size);
239                Node<E> curr = head;
240                int pos = 0;
241                final int newLevel = randomLevel();
242                final Node<E> newNode = new Node<E>(element, newLevel);
243                if (newLevel > level) {
244                        for (int i = level; i < newLevel; i++)
245                                head.dist[i] = size + 1;
246                        level = newLevel;
247                }
248                for (int i = level - 1; i >= 0; i--) {
249                        while (pos + curr.dist[i] <= index) {
250                                pos += curr.dist[i];
251                                curr = curr.next[i];
252                        }
253                        if (i > newLevel - 1)
254                                curr.dist[i]++;
255                        else {
256                                newNode.next[i] = curr.next[i];
257                                curr.next[i] = newNode;
258                                newNode.dist[i] = pos + curr.dist[i] - index;
259                                curr.dist[i] = index + 1 - pos;
260                        }
261                }
262                newNode.prev = curr;
263                newNode.next[0].prev = newNode;
264                modCount++;
265                size++;
266        }
267
268        @Override
269        public E get(int index) {
270                checkElementIndex(index, size);
271                return search(index).element;
272
273        }
274
275        @Override
276        public int lastIndexOf(Object o) {
277                Node<E> node = head;
278                for (int i = size - 1; i >= 0; i--) {
279                        node = node.prev;
280                        if (Objects.equal(node.element, o))
281                                return i;
282                }
283                return -1;
284        }
285
286        @Override
287        public ListIterator<E> listIterator() {
288                return new ListItor();
289        }
290
291        @Override
292        public ListIterator<E> listIterator(int index) {
293                checkPositionIndex(index, size);
294                return new ListItor(index);
295        }
296
297        @SuppressWarnings("unchecked")
298        @Override
299        public E remove(int index) {
300                checkElementIndex(index, size);
301                final Node<E>[] update = new Node[level];
302                Node<E> node = head;
303                int pos = 0;
304                for (int i = level - 1; i >= 0; i--) {
305                        while (pos + node.dist[i] <= index) {
306                                pos += node.dist[i];
307                                node = node.next[i];
308                        }
309                        update[i] = node;
310                }
311                node = node.next[0];
312                delete(node, update);
313                return node.element;
314        }
315
316        @Override
317        public void clear() {
318                for (int i = 0; i < MAX_LEVEL; i++) {
319                        head.next[i] = head;
320                        head.dist[i] = 1;
321                }
322                head.prev = head;
323                modCount++;
324                size = 0;
325        }
326
327        @Override
328        public E set(int index, E element) {
329                checkElementIndex(index, size);
330                Node<E> node = search(index);
331                E e = node.element;
332                node.element = element;
333                return e;
334        }
335
336        /**
337         * Returns a shallow copy of this {@code RankList}. The elements themselves
338         * are not cloned.
339         * 
340         * @return a shallow copy of this list
341         */
342        @SuppressWarnings("unchecked")
343        @Override
344        public RankList<E> clone() {
345                RankList<E> clone;
346                try {
347                        clone = (RankList<E>) super.clone();
348                } catch (java.lang.CloneNotSupportedException e) {
349                        throw new InternalError();
350                }
351                for (int i = 0; i < MAX_LEVEL; i++) {
352                        clone.head.next[i] = clone.head;
353                        clone.head.dist[i] = 1;
354                }
355                clone.head.prev = clone.head;
356                clone.random = new Random();
357                clone.level = 1;
358                clone.modCount = 0;
359                clone.size = 0;
360                clone.addAll(this);
361                return clone;
362        }
363
364        private void writeObject(java.io.ObjectOutputStream oos)
365                        throws java.io.IOException {
366                oos.defaultWriteObject();
367                oos.writeInt(size);
368                for (E e : this)
369                        oos.writeObject(e);
370        }
371
372        @SuppressWarnings("unchecked")
373        private void readObject(java.io.ObjectInputStream ois)
374                        throws java.io.IOException, ClassNotFoundException {
375                ois.defaultReadObject();
376                head = new Node<E>(null, MAX_LEVEL);
377                for (int i = 0; i < MAX_LEVEL; i++) {
378                        head.next[i] = head;
379                        head.dist[i] = 1;
380                }
381                head.prev = head;
382                random = new Random();
383                level = 1;
384                int size = ois.readInt();
385                for (int i = 0; i < size; i++)
386                        add((E) ois.readObject());
387        }

389        private class ListItor implements ListIterator<E> {
390                private Node<E> node;
391                private Node<E> last = null;
392                private int offset;
393                private int index = 0;
394                private int expectedModCount = modCount;
395
396                private ListItor() {
397                        node = head.next[0];
398                        offset = 0;
399                }
400
401                private ListItor(final int index) {
402                        node = search(index);
403                        offset = index;
404                }
405
406                @Override
407                public void add(E element) {
408                        checkForConcurrentModification();
409                        RankList.this.add(index++ + offset, element);
410                        expectedModCount = modCount;
411                        last = null;
412                        node = node.next[0];
413                }
414
415                @Override
416                public boolean hasNext() {
417                        return index + offset < size();
418                }
419
420                @Override
421                public boolean hasPrevious() {
422                        return index + offset > 0;
423                }
424
425                @Override
426                public E next() {
427                        checkForConcurrentModification();
428                        if (!hasNext())
429                                throw new NoSuchElementException();
430                        index++;
431                        last = node;
432                        node = node.next[0];
433                        return last.element;
434                }
435
436                @Override
437                public int nextIndex() {
438                        return index + offset;
439                }
440
441                @Override
442                public E previous() {
443                        checkForConcurrentModification();
444                        if (index == 0)
445                                throw new NoSuchElementException();
446                        index--;
447                        last = node = node.prev;
448                        return node.element;
449                }
450
451                @Override
452                public int previousIndex() {
453                        return index + offset - 1;
454                }
455
456                @Override
457                public void remove() {
458                        checkForConcurrentModification();
459                        checkState(last != null);
460                        RankList.this.remove(--index + offset);
461                        expectedModCount = modCount;
462                        last = null;
463                }
464
465                @Override
466                public void set(E element) {
467                        checkForConcurrentModification();
468                        checkState(last != null);
469                        last.element = element;
470                }
471
472                private void checkForConcurrentModification() {
473                        if (expectedModCount != modCount)
474                                throw new ConcurrentModificationException();
475                }
476        }
477
478        // skip list
479
480        private static class Node<E> {
481                private E element;
482                private Node<E> prev;
483                private final Node<E>[] next;
484                private final int[] dist;
485
486                @SuppressWarnings("unchecked")
487                private Node(final E element, final int size) {
488                        this.element = element;
489                        next = new Node[size];
490                        dist = new int[size];
491                }
492        }
493
494        private Node<E> search(final int index) {
495                Node<E> curr = head;
496                int pos = -1;
497                for (int i = level - 1; i >= 0; i--)
498                        while (pos + curr.dist[i] <= index) {
499                                pos = pos + curr.dist[i];
500                                curr = curr.next[i];
501                        }
502                return curr;
503        }
504
505        private void delete(final Node<E> node, final Node<E>[] update) {
506                node.next[0].prev = node.prev;
507                for (int i = 0; i < level; i++)
508                        if (update[i].next[i] == node) {
509                                update[i].next[i] = node.next[i];
510                                update[i].dist[i] += node.dist[i] - 1;
511                        } else
512                                update[i].dist[i]--;
513                while (head.next[level - 1] == head && level > 1)
514                        level--;
515                modCount++;
516                size--;
517        }
518
519        private int randomLevel() {
520                int randomLevel = 1;
521                while (randomLevel < MAX_LEVEL - 1 && random.nextDouble() < P)
522                        randomLevel++;
523                return randomLevel;
524        }
