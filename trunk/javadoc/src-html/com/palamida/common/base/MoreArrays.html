017package com.palamida.common.base;
018
019import static com.google.common.base.Preconditions.checkNotNull;
020
021import java.util.Arrays;
022
023import com.google.common.base.Function;
024import com.google.common.base.Objects;
025import com.google.common.collect.ObjectArrays;
026
027/**
028 * Static utility methods that operate on or return arrays.
029 * 
030 * @author Zhenya Leonov
031 * @see Arrays
032 * @see ObjectArrays
033 */
034final public class MoreArrays {
035
036        private MoreArrays() {
037        };
038
039        /**
040         * Returns a new array containing the result of {@code function.apply(F)} on
041         * each element of the specified array.
042         * 
043         * @param function
044         *            the function to apply to each element of the specified array
045         * @param from
046         *            the array to transform
047         * @return new array containing the result of {@code function.apply(F)} on
048         *         each element of the specified array
049         */
050        @SuppressWarnings("unchecked")
051        public static <F, T> T[] transform(final F[] from, final Function<? super F, ? extends T> function) {
052                checkNotNull(from);
053                checkNotNull(function);
054                final Object[] to = new Object[from.length];
055                for (int i = 0; i < to.length; i++)
056                        to[i] = function.apply(from[i]);
057                return (T[]) to;
058        }
059
060        /**
061         * Returns {@code true} if the specified array contains the given element,
062         * {@code false} otherwise.
063         * 
064         * @param array
065         *            the specified array
066         * @param element
067         *            the object to look find
068         * @return {@code true} if the specified array contains the given element,
069         *         {@code false} otherwise
070         */
071        public static <T> boolean contains(final T[] array, final Object element) {
072                checkNotNull(array);
073                checkNotNull(element);
074                for (final Object o : array)
075                        if (Objects.equal(o, element))
076                                return true;
077                return false;
078        }
079
080}
