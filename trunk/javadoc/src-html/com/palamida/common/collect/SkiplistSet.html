037/**
038 * <b>Note: This class should only be used for testing. It provides no practical
039 * benefit when compared to {@link java.util.TreeSet java.util.TreeSet} and
040 * lacks navigation methods defined in the {@link NavigableSet} interface.</b>
041 * <p>
042 * A {@code Set} implementation based on a modified <a
043 * href="http://en.wikipedia.org/wiki/Skip_list">skip list</a>. Elements are
044 * sorted from <i>least</i> to <i>greatest</i> according to their <i>natural
045 * ordering</i>, or by an explicit {@link Comparator} provided at creation.
046 * Attempting to remove or insert {@code null} elements is prohibited. Querying
047 * for {@code null} elements is allowed. Inserting non-comparable elements will
048 * result in a {@code ClassCastException}.
049 * <p>
050 * The iterators obtained from the {@link #iterator()} method are
051 * <i>fail-fast</i>. Attempts to modify the elements in this list at any time
052 * after an iterator is created, in any way except through the iterator's own
053 * remove method, will result in a {@code ConcurrentModificationException}.
054 * <p>
055 * This set is not <i>thread-safe</i>. If multiple threads modify this set
056 * concurrently it must be synchronized externally.
057 * <p>
058 * This implementation uses a comparator (whether or not one is explicitly
059 * provided) to perform all element comparisons. Two elements which are deemed
060 * equal by the comparator's {@code compare(E, E)} method are, from the
061 * standpoint of this set, equal.
062 * <p>
063 * Invented by <a href="http://www.cs.umd.edu/~pugh/">Bill Pugh</a> in 1990, A
064 * skip list is a probabilistic data structure for maintaining items in sorted
065 * order. Strictly speaking it is impossible to make any hard guarantees
066 * regarding the worst-case performance of this class. Practical performance is
067 * <i>expected</i> to be logarithmic with an extremely high degree of
068 * probability as the list grows.
069 * <p>
070 * The underlying array-based skip list provides the following expected case
071 * running time (where <i>n</i> is the size of this set and <i>m</i> is the size
072 * of the specified collection which is iterable in linear time):
073 * <p>
074 * <table border="1" cellpadding="3" cellspacing="1" style="width:400px;">
075 *   <tr>
076 *     <th style="text-align:center;">Method</th>
077 *     <th style="text-align:center;">Running Time</th>
078 *   </tr>
079 *   <tr>
080 *     <td>
081 *       {@link #addAll(Collection) addAll(Collection)}<br/>
082 *       {@link #containsAll(Collection) containsAll(Collection)}<br/>
083 *       {@link #retainAll(Collection) retainAll(Collection)}<br/>
084 *       {@link #removeAll(Collection) removeAll(Collection)}
085 *     </td>
086 *     <td style="text-align:center;"><i>O(m log n)</i></td>
087 *   </tr>
088 *   <tr>
089 *     <td>
090 *       {@link #add(Object) add(E)}<br/>
091 *       {@link #contains(Object)}<br/>
092 *       {@link #remove(Object)}<br/>
093 *       {@link Iterator#remove()}
094 *     </td>
095 *     <td style="text-align:center;"><i>O(log n)</i></td>
096 *   </tr>
097 *   <tr>
098 *     <td>
099 *       {@link #clear() clear()}<br/>
100 *       {@link #isEmpty() isEmpty()}<br/>
101 *       {@link #size()}
102 *     </td>
103 *     <td style="text-align:center;"><i>O(1)</i></td>
104 *   </tr>
105 * </table>
106 * 
107 * @author Zhenya Leonov
108 * @param <E>
109 *            the type of elements maintained by this list
110 * @see Skiplist
111 */
112final public class SkiplistSet<E> extends AbstractSet<E> implements
113                SortedCollection<E>, Serializable, Cloneable {
114
115        private static final long serialVersionUID = 1L;
116        private static final double P = .5;
117        private static final int MAX_LEVEL = 32;
118        transient int size = 0;
119        private transient int level = 1;
120        private transient Random random = new Random();
121        private transient Node<E> head = new Node<E>(null, MAX_LEVEL);
122        private final Comparator<? super E> comparator;
123        @SuppressWarnings("unchecked")
124        private transient Node<E>[] update = new Node[MAX_LEVEL];
125        private transient int[] index = new int[MAX_LEVEL];
126        transient int modCount = 0;
127
128        private SkiplistSet(final Comparator<? super E> comparator) {
129                this.comparator = comparator;
130                for (int i = 0; i < MAX_LEVEL; i++)
131                        head.next[i] = head;
132        }
133
134        /**
135         * Creates a new {@code SkiplistSet} that orders its elements according to
136         * their <i>natural ordering</i>.
137         * 
138         * @return a new {@code SkiplistSet} that orders its elements according to
139         *         their <i>natural ordering</i>
140         */
141        public static <E extends Comparable<? super E>> SkiplistSet<E> create() {
142                return new SkiplistSet<E>(Ordering.natural());
143        }
144
145        /**
146         * Creates a new {@code SkiplistSet} that uses the specified comparator to
147         * order its elements.
148         * 
149         * @param comparator
150         *            the specified comparator
151         * @return a new {@code SkiplistSet} that uses the specified comparator to
152         *         order its elements
153         * 
154         */
155        public static <E> SkiplistSet<E> create(
156                        final Comparator<? super E> comparator) {
157                return new SkiplistSet<E>(comparator);
158        }
159
160        /**
161         * Creates a new {@code SkiplistSet} containing the specified initial
162         * elements. If {@code elements} is an instance of {@link SortedSet},
163         * {@link PriorityQueue}, {@link MinMaxPriorityQueue}, or
164         * {@code SortedCollection}, this set will be ordered according to the same
165         * ordering. Otherwise, this set will be ordered according to the <i>natural
166         * ordering</i> of its elements.
167         * 
168         * @param elements
169         *            the collection whose elements are to be placed into the set
170         * @return a new {@code SkiplistSet} containing the elements of the
171         *         specified collection
172         * @throws ClassCastException
173         *             if elements of the specified collection cannot be compared to
174         *             one another according to this set's ordering
175         * @throws NullPointerException
176         *             if any of the elements of the specified collection or the
177         *             collection itself is {@code null}
178         */
179        @SuppressWarnings({ "unchecked" })
180        public static <E> SkiplistSet<E> create(
181                        final Collection<? extends E> elements) {
182                checkNotNull(elements);
183                final Comparator<? super E> comparator;
184                if (elements instanceof SortedSet<?>)
185                        comparator = ((SortedSet<? super E>) elements).comparator();
186                else if (elements instanceof PriorityQueue<?>)
187                        comparator = ((PriorityQueue<? super E>) elements).comparator();
188                else if (elements instanceof SortedCollection<?>)
189                        comparator = ((SortedCollection<? super E>) elements).comparator();
190                else if (elements instanceof MinMaxPriorityQueue<?>)
191                        comparator = ((MinMaxPriorityQueue<? super E>) elements)
192                                        .comparator();
193                else
194                        comparator = (Comparator<? super E>) Ordering.natural();
195                final SkiplistSet<E> set = SkiplistSet.create(comparator);
196                set.addAll(elements);
197                return set;
198        }
199
200        /**
201         * Returns the comparator used to order the elements in this set. If one was
202         * not explicitly provided a <i>natural order</i> comparator is returned.
203         * 
204         * @return the comparator used to order this set
205         */
206        @Override
207        public Comparator<? super E> comparator() {
208                return comparator;
209        }
210
211        /**
212         * Inserts the specified element into this set in sorted order.
213         */
214        @Override
215        public boolean add(E e) {
216                checkNotNull(e);
217                final int newLevel = randomLevel();
218                Node<E> x = head;
219                Node<E> y = head;
220                int i;
221                int idx = 0;
222                for (i = level - 1; i >= 0; i--) {
223                        while (x.next[i] != y
224                                        && comparator.compare(x.next[i].element, e) < 0)
225                                x = x.next[i];
226                        y = x.next[i];
227                        update[i] = x;
228                        index[i] = idx;
229                }
230                if (newLevel > level) {
231                        for (i = level; i < newLevel; i++)
232                                update[i] = head;
233                        level = newLevel;
234                }
235                if (x.next().element != null
236                                && comparator.compare(x.next().element, e) == 0)
237                        return false;
238                x = new Node<E>(e, newLevel);
239                for (i = 0; i < level; i++)
240                        if (i < newLevel) {
241                                x.next[i] = update[i].next[i];
242                                update[i].next[i] = x;
243                        }
244                modCount++;
245                size++;
246                return true;
247        }
248
249        @Override
250        @SuppressWarnings("unchecked")
251        public boolean contains(Object o) {
252                return o != null && search((E) o) != null;
253        }
254
255        @Override
256        public Iterator<E> iterator() {
257                return new IteratorImpl();
258        }
259
260        @SuppressWarnings("unchecked")
261        @Override
262        public boolean remove(Object o) {
263                checkNotNull(o);
264                final E element = (E) o;
265                Node<E> curr = head;
266                for (int i = level - 1; i >= 0; i--) {
267                        while (curr.next[i] != head
268                                        && comparator.compare(curr.next[i].element, element) < 0)
269                                curr = curr.next[i];
270                        update[i] = curr;
271                }
272                curr = curr.next();
273                if (curr == head || comparator.compare(curr.element, element) != 0)
274                        return false;
275                delete(curr, update);
276                return true;
277        }
278
279        @Override
280        public int size() {
281                return size;
282        }
283
284        @Override
285        public void clear() {
286                for (int i = 0; i < MAX_LEVEL; i++)
287                        head.next[i] = head;
288                modCount++;
289                size = 0;
290        }
291
292        /**
293         * Returns a shallow copy of this {@code SkiplistSet}. The elements
294         * themselves are not cloned.
295         * 
296         * @return a shallow copy of this set
297         */
298        @SuppressWarnings("unchecked")
299        @Override
300        public SkiplistSet<E> clone() throws CloneNotSupportedException {
301                SkiplistSet<E> clone;
302                try {
303                        clone = (SkiplistSet<E>) super.clone();
304                } catch (java.lang.CloneNotSupportedException e) {
305                        throw new InternalError();
306                }
307                for (int i = 0; i < MAX_LEVEL; i++)
308                        clone.head.next[i] = clone.head;
309                clone.random = new Random();
310                clone.level = 1;
311                clone.modCount = 0;
312                clone.size = 0;
313                clone.addAll(this);
314                return clone;
315        }
316
317        private void writeObject(java.io.ObjectOutputStream oos)
318                        throws java.io.IOException {
319                oos.defaultWriteObject();
320                oos.writeInt(size);
321                for (E e : this)
322                        oos.writeObject(e);
323        }
324
325        @SuppressWarnings("unchecked")
326        private void readObject(java.io.ObjectInputStream ois)
327                        throws java.io.IOException, ClassNotFoundException {
328                ois.defaultReadObject();
329                head = new Node<E>(null, MAX_LEVEL);
330                for (int i = 0; i < MAX_LEVEL; i++)
331                        head.next[i] = head;
332                update = new Node[MAX_LEVEL];
333                index = new int[MAX_LEVEL];
334                random = new Random();
335                level = 1;
336                int size = ois.readInt();
337                for (int i = 0; i < size; i++)
338                        add((E) ois.readObject());
339        }
340
341        private class IteratorImpl implements Iterator<E> {
342                private Node<E> node = head.next();
343                private Node<E> last = null;
344                private int index = 0;
345                private int expectedModCount = modCount;
346
347                @Override
348                public boolean hasNext() {
349                        return index < size();
350                }
351
352                @Override
353                public E next() {
354                        checkForConcurrentModification();
355                        if (!hasNext())
356                                throw new NoSuchElementException();
357                        index++;
358                        last = node;
359                        node = node.next();
360                        return last.element;
361                }
362
363                @Override
364                public void remove() {
365                        checkForConcurrentModification();
366                        checkState(last != null);
367                        SkiplistSet.this.remove(last.element);
368                        expectedModCount = modCount;
369                        last = null;
370                }
371
372                void checkForConcurrentModification() {
373                        if (expectedModCount != modCount)
374                                throw new ConcurrentModificationException();
375                }
376        }
377
378        // skip set
379
380        private static class Node<E> {
381                private E element;
382                private final Node<E>[] next;
383
384                @SuppressWarnings("unchecked")
385                private Node(final E element, final int size) {
386                        this.element = element;
387                        next = new Node[size];
388                }
389
390                private Node<E> next() {
391                        return next[0];
392                }
393        }
394
395        private int randomLevel() {
396                int randomLevel = 1;
397                while (randomLevel < MAX_LEVEL - 1 && random.nextDouble() < P)
398                        randomLevel++;
399                return randomLevel;
400        }
401
402        private void delete(final Node<E> node, final Node<E>[] update) {
403                for (int i = 0; i < level; i++)
404                        if (update[i].next[i] == node) {
405                                update[i].next[i] = node.next[i];
406                        }
407                while (head.next[level - 1] == head && level > 1)
408                        level--;
409                modCount++;
410                size--;
411        }
412
413        private Node<E> search(final E element) {
414                Node<E> curr = head;
415                for (int i = level - 1; i >= 0; i--)
416                        while (curr.next[i] != head
417                                        && comparator.compare(curr.next[i].element, element) < 0)
418                                curr = curr.next[i];
419                curr = curr.next();
420                if (curr != head && comparator.compare(curr.element, element) == 0)
421                        return curr;
422                return null;
423        }
