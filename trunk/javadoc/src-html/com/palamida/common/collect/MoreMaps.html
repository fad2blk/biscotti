001package com.palamida.common.collect;
002
003import static com.google.common.base.Preconditions.checkArgument;
004import static com.google.common.base.Preconditions.checkNotNull;
005
006import java.util.LinkedHashMap;
007import java.util.Map;
008
009import com.google.common.collect.Maps;
010
011/**
012 * Static utility methods which operate on or return {@link Map}s.
013 * 
014 * @author Zhenya Leonov
015 * @see Maps
016 */
017public class MoreMaps {
018
019        private MoreMaps() {
020        }
021
022        /**
023         * Creates an empty {@code LinkedHashMap} which orders its keys according to
024         * their <i>access-order</i>.
025         * 
026         * @return an empty {@code LinkedHashMap} which orders its keys according to
027         *         their <i>access-order</i>
028         * @see LinkedHashMap
029         */
030        public static <K, V> Map<K, V> newAccessOrderMap() {
031                return new LinkedHashMap<K, V>(16, .75F, true);
032        }
033
034        /**
035         * Creates a {@code LinkedHashMap} which orders its keys according to their
036         * <i>access-order</i>, containing the same mappings as the specified map.
037         * 
038         * @param m
039         *            the map whose mappings this map should contain
040         * @return a {@code LinkedHashMap} which orders its keys according to their
041         *         <i>access-order</i>, containing the same mappings as the
042         *         specified map
043         * @see LinkedHashMap
044         */
045        public static <K, V> Map<K, V> newAccessOrderMap(
046                        final Map<? extends K, ? extends V> m) {
047                checkNotNull(m);
048                final Map<K, V> map = new LinkedHashMap<K, V>(Math.max(
049                                (int) (m.size() / .75F) + 1, 16), .75F, true);
050                map.putAll(m);
051                return map;
052        }
053
054        /**
055         * Creates an empty {@code LinkedHashMap} which orders its keys according to
056         * their <i>access-order</i>, with enough capacity to hold the specified
057         * number of entries without rehashing.
058         * 
059         * @param expectedSize
060         *            the expected size
061         * @return an empty {@code LinkedHashMap} which orders its keys according to
062         *         their <i>access-order</i>, with enough capacity to hold the
063         *         specified number of entries without rehashing
064         * @see LinkedHashMap
065         */
066        public static <K, V> Map<K, V> newAccessOrderMapWithExpectedSize(
067                        final int expectedSize) {
068                checkArgument(expectedSize >= 0);
069                return new LinkedHashMap<K, V>(Math.max(expectedSize * 2, 16), .75F,
070                                true);
071        }
072
073}
