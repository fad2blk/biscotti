019/**
020 * An implementation of {@link BoundedQueue} backed by a {@link TreeQueue}. The
021 * elements of this queue are sorted according to their <i>natural ordering</i>,
022 * or by an explicit {@link Comparator} provided at creation. Inserting
023 * {@code null} elements is prohibited. Attempting to insert non-comparable
024 * elements will result in a {@code ClassCastException}.
025 * <p>
026 * The first element (the head) of this queue is considered to be the
027 * <i>least</i> element with respect to the specified ordering. Elements with
028 * equal priority are ordered according to their insertion order.
029 * <p>
030 * When the queue is full the {@code offer(E)} method behaves according to the
031 * following policy: if the element to be added has higher priority than the
032 * lowest priority element currently in the queue, the new element is added and
033 * the lowest priority element is removed; else the new element is rejected. The
034 * backing {@code TreeQueue} maintains references to the highest and lowest
035 * elements in the queue; rejecting an element is an <i>O(1)</i> operation.
036 * <p>
037 * The {@code add(E)} and {@code addAll(Collection)} operations will throw an
038 * {@code IllegalStateException} when the queue is full and a new element is
039 * rejected; as required by the contract of {@link Queue#add Queue.add(E)}.
040 * <p>
041 * Bounded priority queues are useful when implementing <i>n-best</i> algorithms
042 * (e.g. finding the <i>best</i> <i>n</i> elements in an arbitrary collection).
043 * <p>
044 * This queue is not <i>thread-safe</i>. If multiple threads modify this queue
045 * concurrently it must be synchronized externally.
046 * <p>
047 * Refer to {@link TreeQueue} for details of the underlying implementation.
048 * 
049 * @author Zhenya Leonov
050 * @param <E>
051 *            the type of elements held in this queue
052 */
053final public class TreeBoundedQueue<E> extends ForwardingQueue<E> implements
054                BoundedQueue<E>, SortedCollection<E>, Serializable, Cloneable {
055
056        private static final long serialVersionUID = 1L;
057        private TreeQueue<E> delegate;
058        private final int maximumSize;
059
060        private TreeBoundedQueue(final int maximumSize,
061                        final Comparator<? super E> comparator) {
062                delegate = TreeQueue.orderedBy(comparator).create();
063                this.maximumSize = maximumSize;
064        }
065
066        private TreeBoundedQueue(final Comparator<? super E> comparator,
067                        final Iterable<? extends E> elements) {
068                delegate = TreeQueue.orderedBy(comparator).create();
069                for (E e : elements)
070                        offer(e);
071                checkArgument(size() > 0);
072                this.maximumSize = size();
073        }
074
075        /**
076         * Creates a new {@code TreeBoundedQeque} which orders its elements
077         * according to their <i>natural ordering</i>, and having the specified
078         * maximum size.
079         * 
080         * @param maximumSize
081         *            the maximum size (the bound) of this queue
082         * @return returns a new {@code TreeBoundedQeque} which orders its elements
083         *         according to their <i>natural ordering</i>, having the specified
084         *         maximum size
085         * @throws IllegalArgumentException
086         *             if {@code maxSize} is less than 1
087         */
088        public static <E extends Comparable<? super E>> TreeBoundedQueue<E> create(
089                        final int maximumSize) {
090                checkArgument(maximumSize > 0);
091                return new TreeBoundedQueue<E>(maximumSize, Ordering.natural());
092        }

115        /**
116         * Creates a new {@code TreeBoundedQueue} containing the elements of, and
117         * having the maximum size equal to the number of elements in the specified
118         * collection. If the collection is an instance of {@link SortedSet},
119         * {@link PriorityQueue}, {@link MinMaxPriorityQueue}, or
120         * {@code SortedCollection} this queue will be ordered according to the same
121         * ordering. Otherwise, this queue will be ordered according to the
122         * <i>natural ordering</i> of its elements.
123         * 
124         * @param elements
125         *            the collection whose elements are to be placed into the queue
126         * @return a new {@code TreeBoundedQueue} containing the elements of the
127         *         specified collection
128         * @throws ClassCastException
129         *             if elements of the specified collection cannot be compared to
130         *             one another according to the priority queue's ordering
131         * @throws NullPointerException
132         *             if any of the elements of the specified collection or the
133         *             collection itself is {@code null}
134         * @throws IllegalArgumentException
135         *             if the specified collection is empty
136         */
137        @SuppressWarnings({ "unchecked" })
138        public static <E> TreeBoundedQueue<E> create(
139                        final Collection<? extends E> elements) {
140                checkNotNull(elements);
141                final Comparator<? super E> comparator;
142                if (elements instanceof SortedSet<?>)
143                        comparator = ((SortedSet<? super E>) elements).comparator();
144                else if (elements instanceof PriorityQueue<?>)
145                        comparator = ((PriorityQueue<? super E>) elements).comparator();
146                else if (elements instanceof SortedCollection<?>)
147                        comparator = ((SortedCollection<? super E>) elements).comparator();
148                else if (elements instanceof MinMaxPriorityQueue<?>)
149                        comparator = ((MinMaxPriorityQueue<? super E>) elements)
150                                        .comparator();
151                else
152                        comparator = (Comparator<? super E>) Ordering.natural();
153                return new TreeBoundedQueue<E>(comparator, elements);
154        }
155
156        /**
157         * Returns a new builder configured to build {@code TreeBoundedQueue}
158         * instances that use the specified comparator for ordering.
159         * 
160         * @param comparator
161         *            the specified comparator
162         * @return a new builder configured to build {@code TreeBoundedQueue}
163         *         instances that use the specified comparator for ordering
164         */
165        public static <B> Builder<B> orderedBy(final Comparator<B> comparator) {
166                checkNotNull(comparator);
167                return new Builder<B>(comparator);
168        }
169
170        /**
171         * A builder for the creation of {@code TreeBoundedQueue} instances.
172         * Instances of this builder are obtained calling
173         * {@link TreeBoundedQueue#orderedBy(Comparator)}.
174         * 
175         * @author Zhenya Leonov
176         * @param <B>
177         *            the upper bound of the type of queues this builder can produce
178         *            (for example a {@code Builder<Number>} can produce a
179         *            {@code TreeBoundedQueue<Float>} or a
180         *            {@code TreeBoundedQueue<Integer>}
181         */
182        public static final class Builder<B> {
183
184                private final Comparator<B> comparator;
185
186                private Builder(final Comparator<B> comparator) {
187                        this.comparator = comparator;
188                }
189
190                /**
191                 * Builds an empty {@code TreeBoundedQueue} using the previously
192                 * specified comparator and the given maximum size.
193                 * 
194                 * @param maximumSize
195                 *            the maximum size of this queue
196                 * @return an empty {@code TreeBoundedQueue} using the previously
197                 *         specified comparator and the given maximum size
198                 * @throws IllegalArgumentException
199                 *             if the maximum size has not been specified
200                 */
201                public <T extends B> TreeBoundedQueue<T> create(final int maximumSize) {
202                        checkArgument(maximumSize > 0, "maximum size < 1");
203                        return new TreeBoundedQueue<T>(maximumSize, comparator);
204                }
205
206                /**
207                 * Builds a new {@code TreeBoundedQueue} using the previously specified
208                 * comparator and maximum size, having the given initial elements.
209                 * 
210                 * @param maximumSize
211                 *            the maximum size of this queue
212                 * @param elements
213                 *            the initial elements to be placed in this queue
214                 * @return a new {@code TreeBoundedQueue} using the previously specified
215                 *         comparator and maximum size, having the given initial
216                 *         elements
217                 * @throws IllegalStateException
218                 *             if the maximum size has not been specified
219                 */
220                public <T extends B> TreeBoundedQueue<T> create(final int maximumSize,
221                                final Iterable<? extends T> elements) {
222                        checkArgument(maximumSize > 0, "maximum size < 1");
223                        checkNotNull(elements);
224                        final TreeBoundedQueue<T> list = new TreeBoundedQueue<T>(
225                                        maximumSize, comparator);
226                        Iterables.addAll(list, elements);
227                        return list;
228                }
229        }
230
231        @Override
232        public boolean add(E e) {
233                checkState(offer(e), "Queue full");
234                return true;
235        }
236
237        @Override
238        public boolean offer(E e) {
239                if (size() == maximumSize)
240                        if (comparator().compare(e, delegate().peekLast()) < 0)
241                                delegate().pollLast();
242                        else
243                                return false;
244                return delegate().offer(e);
245        }
246
247        @Override
248        public int capacity() {
249                return maximumSize;
250        }
251
252        @Override
253        public int remainingCapacity() {
254                return maximumSize - size();
255        }
256
257        @Override
258        public Comparator<? super E> comparator() {
259                return delegate().comparator();
260        }
261
262        @Override
263        protected TreeQueue<E> delegate() {
264                return delegate;
265        }
266
267        /**
268         * Returns a shallow copy of this {@code TreeBoundedQueue}. The elements
269         * themselves are not cloned.
270         * 
271         * @return a shallow copy of this queue
272         */
273        @SuppressWarnings("unchecked")
274        @Override
275        public TreeBoundedQueue<E> clone() throws CloneNotSupportedException {
276                TreeBoundedQueue<E> clone = (TreeBoundedQueue<E>) super.clone();
277                clone.delegate = (TreeQueue<E>) delegate.clone();
278                return clone;
279        }
