043/**
044 * A {@link Sortedlist} implementation, based on a modified <a
045 * href="http://en.wikipedia.org/wiki/Red-black_tree">Red-Black Tree</a>.
046 * Elements are sorted from <i>least</i> to <i>greatest</i> according to their
047 * <i>natural ordering</i>, or by an explicit {@link Comparator} provided at
048 * creation. Attempting to remove or insert {@code null} elements is prohibited.
049 * Querying for {@code null} elements is allowed. Inserting non-comparable
050 * elements will result in a {@code ClassCastException}.
051 * <p>
052 * The iterators obtained from the {@link #iterator()} and
053 * {@link #listIterator()} methods are <i>fail-fast</i>. Attempts to modify the
054 * elements in this list at any time after an iterator is created, in any way
055 * except through the iterator's own remove method, will result in a
056 * {@code ConcurrentModificationException}. Further, the list iterator does not
057 * support the {@code add(E)} and {@code set(E)} operations.
058 * <p>
059 * This sorted-list is not <i>thread-safe</i>. If multiple threads modify this
060 * sorted-list concurrently it must be synchronized externally.
061 * <p>
062 * This implementation uses a comparator (whether or not one is explicitly
063 * provided) to perform all element comparisons. Two elements which are deemed
064 * equal by the comparator's {@code compare(E, E)} method are, from the
065 * standpoint of this list, equal. Further, no guarantee is made as to the final
066 * order of <i>equal</i> elements. Ties may be broken arbitrarily.
067 * <p>
068 * The underlying Red-Black Tree provides the following worst case running time
069 * (where <i>n</i> is the size of this sorted-list and <i>m</i> is the size of
070 * the specified collection which is iterable in linear time):
071 * <p>
072 * <table border="1" cellpadding="3" cellspacing="1" style="width:400px;">
073 *   <tr>
074 *     <th style="text-align:center;">Method</th>
075 *     <th style="text-align:center;">Running Time</th>
076 *   </tr>
077 *   <tr>
078 *     <td>
079 *       {@link #addAll(Collection) addAll(Collection)}<br/>
080 *       {@link #containsAll(Collection) containsAll(Collection)}<br/>
081 *       {@link #retainAll(Collection) retainAll(Collection)}<br/>
082 *       {@link #removeAll(Collection) removeAll(Collection)}
083 *     </td>
084 *     <td style="text-align:center;"><i>O(m log n)</i></td>
085 *   </tr>
086 *   <tr>
087 *     <td>
088 *       {@link #indexOf(Object)}<br/>
089 *       {@link #lastIndexOf(Object)}<br/>
090 *       {@link #get(int)}<br/>
091 *       {@link #remove(int)}<br/>
092 *       {@link #listIterator(int)}
093 *     </td>
094 *     <td style="text-align:center;"><i>O(n)</i></td>
095 *   </tr>
096 *   <tr>
097 *     <td>
098 *       {@link #add(Object) add(E)}<br/>
099 *       {@link #contains(Object)}<br/>
100 *       {@link #remove(Object)}
101 *     </td>
102 *     <td style="text-align:center;"><i>O(log n)</i></td>
103 *   </tr>
104 *   <tr>
105 *     <td>
106 *       {@link #clear() clear()}<br/>
107 *       {@link #isEmpty() isEmpty()}<br/>
108 *       {@link #size()}<br/>
109 *       {@link Iterator#remove()}<br/>
110 *       {@link ListIterator#remove()}
111 *     </td>
112 *     <td style="text-align:center;"><i>O(1)</i></td>
113 *   </tr>
114 * </table>
115 * <p>
116 * The sub-list views exhibit identical time complexity, with the exception of
117 * the {@code clear()} operation which runs in linear time proportional to the
118 * size of the views.
119 * 
120 * @author Zhenya Leonov
121 * @param <E>
122 *            the type of elements maintained by this list
123 * @see Skiplist
124 */
125public class Treelist<E> extends AbstractCollection<E> implements
126                Sortedlist<E>, Cloneable, Serializable {

152        /**
153         * Creates a new {@code Treelist} containing the specified initial elements.
154         * If {@code elements} is an instance of {@link SortedSet},
155         * {@link PriorityQueue}, {@link MinMaxPriorityQueue}, or
156         * {@code SortedCollection}, this list will be ordered according to the same
157         * ordering. Otherwise, this list will be ordered according to the
158         * <i>natural ordering</i> of its elements.
159         * 
160         * @param elements
161         *            the collection whose elements are to be placed into the list
162         * @return a new {@code Treelist} containing the elements of the specified
163         *         collection
164         * @throws ClassCastException
165         *             if elements of the specified collection cannot be compared to
166         *             one another according to this list's ordering
167         * @throws NullPointerException
168         *             if any of the elements of the specified collection or the
169         *             collection itself is {@code null}
170         */
171        @SuppressWarnings({ "unchecked" })
172        public static <E> Treelist<E> from(final Collection<? extends E> elements) {
173                checkNotNull(elements);
174                final Comparator<? super E> comparator;
175                if (elements instanceof SortedSet<?>)
176                        comparator = ((SortedSet<? super E>) elements).comparator();
177                else if (elements instanceof PriorityQueue<?>)
178                        comparator = ((PriorityQueue<? super E>) elements).comparator();
179                else if (elements instanceof SortedCollection<?>)
180                        comparator = ((SortedCollection<? super E>) elements).comparator();
181                else if (elements instanceof MinMaxPriorityQueue<?>)
182                        comparator = ((MinMaxPriorityQueue<? super E>) elements)
183                                        .comparator();
184                else
185                        comparator = (Comparator<? super E>) Ordering.natural();
186                return orderedBy(comparator).create(elements);
187        }

288        @Override
289        public int indexOf(Object o) {
290                if (o != null) {
291                        @SuppressWarnings("unchecked")
292                        E e = (E) o;
293                        ListIterator<E> itor = listIterator();
294                        while (itor.hasNext())
295                                if (comparator.compare(itor.next(), e) == 0)
296                                        return itor.previousIndex();
297                }
298                return -1;
299        }

301        @Override
302        public int lastIndexOf(Object o) {
303                if (o != null) {
304                        @SuppressWarnings("unchecked")
305                        E e = (E) o;
306                        ListIterator<E> itor = listIterator();
307                        while (itor.hasNext())
308                                if (comparator.compare(itor.next(), e) == 0) {
309                                        while (itor.hasNext()
310                                                        && comparator.compare(itor.next(), e) == 0)
311                                                ;
312                                        return itor.previousIndex();
313                                }
314                }
315                return -1;
316        }

323        @Override
324        public ListIterator<E> listIterator() {
325                return new ListIterator<E>() {
326                        private int index = 0;
327                        private Node next = min;
328                        private Node prev = nil;
329                        private Node last = nil;
330                        private int expectedModCount = modCount;
331
332                        @Override
333                        public void add(E e) {
334                                throw new UnsupportedOperationException();
335                        }
336
337                        @Override
338                        public boolean hasNext() {
339                                return index < size();
340                        }
341
342                        @Override
343                        public boolean hasPrevious() {
344                                return index > 0;
345                        }
346
347                        @Override
348                        public E next() {
349                                checkForConcurrentModification();
350                                if (index == size())
351                                        throw new NoSuchElementException();
352                                prev = next;
353                                index++;
354                                next = successor(prev);
355                                last = prev;
356                                return prev.element;
357                        }
358
359                        @Override
360                        public int nextIndex() {
361                                return index;
362                        }
363
364                        @Override
365                        public E previous() {
366                                checkForConcurrentModification();
367                                if (index == 0)
368                                        throw new NoSuchElementException();
369                                next = prev;
370                                index--;
371                                prev = predecessor(next);
372                                last = next;
373                                return next.element;
374                        }
375
376                        @Override
377                        public int previousIndex() {
378                                return index - 1;
379                        }
380
381                        @Override
382                        public void remove() {
383                                checkForConcurrentModification();
384                                checkState(last != nil);
385                                if (last.left != nil && last.right != nil)
386                                        next = last;
387                                delete(last);
388                                index--;
389                                expectedModCount = modCount;
390                                last = nil;
391                        }
392
393                        @Override
394                        public void set(E e) {
395                                throw new UnsupportedOperationException();
396                        }
397
398                        private void checkForConcurrentModification() {
399                                if (expectedModCount != modCount)
400                                        throw new ConcurrentModificationException();
401                        }
402                };
403        }

456        @Override
457        public boolean equals(Object o) {
458                if (o == this)
459                        return true;
460                if (!(o instanceof Sortedlist))
461                        return false;
462                try {
463                        @SuppressWarnings("unchecked")
464                        final Iterator<E> i = ((Collection<E>) o).iterator();
465                        for (E e : this)
466                                if (comparator.compare(e, i.next()) != 0)
467                                        return false;
468                        return !i.hasNext();
469                } catch (ClassCastException e) {
470                        return false;
471                } catch (NullPointerException e) {
472                        return false;
473                }
474        }

531        @SuppressWarnings("serial")
532        private class Sublist extends Treelist<E> {
533                private final Treelist<E> list;
534                private final int offset;
535                private Node from;
536                private Node to;
537
538                private void checkForConcurrentModification() {
539                        if (this.modCount != list.modCount)
540                                throw new ConcurrentModificationException();
541                }
542
543                public Sublist(Treelist<E> list, int fromIndex, int toIndex) {
544                        super(list.comparator);
545                        this.list = list;
546                        from = list.min;
547                        offset = fromIndex;
548                        this.modCount = list.modCount;
549                        this.size = toIndex - fromIndex;
550                        int i = 0;
551                        for (; i < fromIndex; i++)
552                                from = successor(from);
553                        to = from;
554                        for (; i < toIndex - 1; i++)
555                                to = successor(to);
556                }
557
558                @Override
559                public boolean add(E e) {
560                        checkForConcurrentModification();
561                        if (comparator.compare(e, from.element) < 0
562                                        || comparator.compare(e, to.element) > 0)
563                                throw new IllegalArgumentException("element out of range");
564                        list.add(e);
565                        this.modCount = list.modCount;
566                        this.size++;
567                        if (comparator.compare(to.element, e) <= 0)
568                                to = successor(to);
569                        return true;
570                }
571
572                @SuppressWarnings("unchecked")
573                @Override
574                public boolean contains(Object o) {
575                        checkForConcurrentModification();
576                        return o != null && search((E) o) != null;
577                }
578
579                @Override
580                public E get(int index) {
581                        checkForConcurrentModification();
582                        checkElementIndex(index, this.size);
583                        return list.get(index + offset);
584                }
585
586                @Override
587                public ListIterator<E> listIterator() {
588                        return listIterator(0);
589                }
590
591                @Override
592                public ListIterator<E> listIterator(final int index) {
593                        checkForConcurrentModification();
594                        checkPositionIndex(index, Sublist.this.size);
595                        return new ListIterator<E>() {
596                                private ListIterator<E> i = list.listIterator(index + offset);
597
598                                @Override
599                                public boolean hasNext() {
600                                        return nextIndex() < Sublist.this.size;
601                                }
602
603                                @Override
604                                public E next() {
605                                        if (hasNext())
606                                                return i.next();
607                                        else
608                                                throw new NoSuchElementException();
609                                }
610
611                                @Override
612                                public boolean hasPrevious() {
613                                        return previousIndex() >= 0;
614                                }
615
616                                @Override
617                                public E previous() {
618                                        if (hasPrevious())
619                                                return i.previous();
620                                        else
621                                                throw new NoSuchElementException();
622                                }
623
624                                @Override
625                                public int nextIndex() {
626                                        return i.nextIndex() - offset;
627                                }
628
629                                @Override
630                                public int previousIndex() {
631                                        return i.previousIndex() - offset;
632                                }
633
634                                @Override
635                                public void remove() {
636                                        i.remove();
637                                        Sublist.this.modCount = list.modCount;
638                                        Sublist.this.size--;
639                                }
640
641                                @Override
642                                public void set(E e) {
643                                        throw new UnsupportedOperationException();
644                                }
645
646                                @Override
647                                public void add(E e) {
648                                        throw new UnsupportedOperationException();
649                                }
650                        };
651                }
652
653                @SuppressWarnings("unchecked")
654                @Override
655                public boolean remove(Object o) {
656                        checkForConcurrentModification();
657                        checkNotNull(o);
658                        final Node node = search((E) o);
659                        if (node == null)
660                                return false;
661                        if (node == to)
662                                to = predecessor(to);
663                        if (node == from)
664                                from = successor(from);
665                        list.delete(node);
666                        this.modCount = list.modCount;
667                        this.size--;
668                        return true;
669                }
670
671                @Override
672                public E remove(int index) {
673                        checkForConcurrentModification();
674                        checkElementIndex(index, this.size);
675                        if (index == 0)
676                                from = successor(from);
677                        if (index == this.size - 1)
678                                to = predecessor(to);
679                        final E e = list.remove(index + offset);
680                        this.modCount = list.modCount;
681                        this.size--;
682                        return e;
683                }
684
685                @Override
686                public int size() {
687                        checkForConcurrentModification();
688                        return this.size;
689                }
690
691                @Override
692                public void clear() {
693                        checkForConcurrentModification();
694                        final Iterator<E> iterator = iterator();
695                        while (iterator.hasNext())
696                                iterator.remove();
697                }

699                @Override
700                public Treelist<E> clone() throws CloneNotSupportedException {
701                        throw new CloneNotSupportedException();
702                }

716                @Override
717                Node search(final E e) {
718                        final int compareFrom = comparator.compare(e, from.element);
719                        final int compareTo = comparator.compare(e, to.element);
720                        if (compareFrom < 0 || compareTo > 0)
721                                return null;
722                        if (compareFrom == 0)
723                                return from;
724                        else if (compareTo == 0)
725                                return to;
726                        else
727                                return list.search(e);
728                }
729        }

733        static enum Color {
734                BLACK, RED;
735        }

737        private class Node {
738                private E element = null;
739                private Node parent, left, right;
740                private Color color = BLACK;
741
742                private Node() {
743                        parent = this;
744                        right = this;
745                        left = this;
746                }
747
748                private Node(final E element) {
749                        this.element = element;
750                        parent = nil;
751                        right = nil;
752                        left = nil;
753                }
754        }
755
756        Node search(final E e) {
757                Node n = root;
758                while (n != nil) {
759                        int cmp = comparator.compare(e, n.element);
760                        if (cmp == 0)
761                                return n;
762                        if (cmp < 0)
763                                n = n.left;
764                        else
765                                n = n.right;
766                }
767                return null;
768        }

770        /**
771         * Introduction to Algorithms (CLR) Second Edition
772         * 
773         * <pre>
774         * RB-INSERT(T, z)
775         * y = nil[T]
776         * x = root[T]
777         * while x != nil[T]
778         *    do y = x
779         *       if key[z] < key[x]
780         *          then x = left[x]
781         *          else x = right[x]
782         * p[z] = y
783         * if y = nil[T]
784         *    then root[T] = z
785         *    else if key[z] < key[y]
786         *            then left[y] = z
787         *            else right[y] = z
788         * left[z] = nil[T]
789         * right[z] = nil[T]
790         * color[z] = RED
791         * RB-INSERT-FIXUP(T, z)
792         */
793        private void insert(Node z) {
794                size++;
795                modCount++;
796                Node x = root;
797                Node y = nil;
798                while (x != nil) {
799                        y = x;
800                        if (comparator.compare(z.element, x.element) < 0)
801                                x = x.left;
802                        else
803                                x = x.right;
804                }
805                z.parent = y;
806                if (y == nil)
807                        root = z;
808                else if (comparator.compare(z.element, y.element) < 0)
809                        y.left = z;
810                else
811                        y.right = z;
812                fixAfterInsertion(z);
813                if (max == nil || comparator.compare(z.element, max.element) >= 0)
814                        max = z;
815                if (min == nil || comparator.compare(z.element, min.element) < 0)
816                        min = z;
817        }

819        /**
820         * Introduction to Algorithms (CLR) Second Edition
821         * 
822         * <pre>
823         * RB-DELETE-FIXUP(T, z)
824         * if left[z] = nil[T] or right[z] = nil[T]
825         *    then y = z
826         *    else y = TREE-SUCCESSOR(z)
827         * if left[y] != nil[T]
828         *    then x = left[y]
829         *    else x = right[y]
830         * p[x] = p[y]
831         * if p[y] = nil[T]
832         *    then root[T] = x
833         *    else if y = left[p[y]]
834         *            then left[p[y]] = x
835         *            else right[p[y]] = x
836         * if y != z
837         *    then key[z] = key[y]
838         *         copy y's satellite data into z
839         * if color[y] = BLACK
840         *    then RB-DELETE-FIXUP(T, x)
841         * return y
842         */
843        private void delete(Node z) {
844                size--;
845                modCount++;
846                Node x, y;
847                if (min == z)
848                        min = successor(z);
849                if (max == z)
850                        max = predecessor(z);
851                if (z.left == nil || z.right == nil)
852                        y = z;
853                else
854                        y = successor(z);
855                if (y.left != nil)
856                        x = y.left;
857                else
858                        x = y.right;
859                x.parent = y.parent;
860                if (y.parent == nil)
861                        root = x;
862                else if (y == y.parent.left)
863                        y.parent.left = x;
864                else
865                        y.parent.right = x;
866                if (y != z)
867                        z.element = y.element;
868                if (y.color == Color.BLACK)
869                        fixAfterDeletion(x);
870        }

872        /**
873         * Introduction to Algorithms (CLR) Second Edition
874         * 
875         * <pre>
876         * TREE-SUCCESSOR(x)
877         * if right[x] != NIL
878         *    then return TREE-MINIMUM(right[x])
879         * y = p[x]
880         * while y != NIL and x = right[y]
881         *    do x = y
882         *       y = p[y]
883         * return y
884         */
885        private Node successor(Node x) {
886                if (x == nil)
887                        return nil;
888                if (x.right != nil) {
889                        Node y = x.right;
890                        while (y.left != nil)
891                                y = y.left;
892                        return y;
893                }
894                Node y = x.parent;
895                while (y != nil && x == y.right) {
896                        x = y;
897                        y = y.parent;
898                }
899                return y;
900        }
901
902        private Node predecessor(Node x) {
903                if (x == nil)
904                        return nil;
905                if (x.left != nil) {
906                        Node y = x.left;
907                        while (y.right != nil)
908                                y = y.right;
909                        return y;
910                }
911                Node y = x.parent;
912                while (y != nil && x == y.left) {
913                        x = y;
914                        y = y.left;
915                }
916                return y;
917        }

919        /**
920         * Introduction to Algorithms (CLR) Second Edition
921         * 
922         * <pre>
923         * LEFT-ROTATE(T, x)
924         * y = right[x]                                                 Set y.
925         * right[x] = left[y]                                   Turn y's left subtree into x's right subtree.
926         * if left[y] != nil[T]
927         *    then p[left[y]] = x
928         * p[y] = p[x]                                                  Link x's parent to y.
929         * if p[x] = nil[T]
930         *    then root[T] = y
931         *    else if x = left[p[x]]
932         *            then left[p[x]] = y
933         *            else right[p[x]] = y
934         * left[y] = x                                                  Put x on y's left.
935         * p[x] = y
936         */
937        private void leftRotate(final Node x) {
938                if (x != nil) {
939                        Node n = x.right;
940                        x.right = n.left;
941                        if (n.left != nil)
942                                n.left.parent = x;
943                        n.parent = x.parent;
944                        if (x.parent == nil)
945                                root = n;
946                        else if (x.parent.left == x)
947                                x.parent.left = n;
948                        else
949                                x.parent.right = n;
950                        n.left = x;
951                        x.parent = n;
952                }
953        }
954
955        private void rightRotate(final Node x) {
956                if (x != nil) {
957                        Node n = x.left;
958                        x.left = n.right;
959                        if (n.right != nil)
960                                n.right.parent = x;
961                        n.parent = x.parent;
962                        if (x.parent == nil)
963                                root = n;
964                        else if (x.parent.right == x)
965                                x.parent.right = n;
966                        else
967                                x.parent.left = n;
968                        n.right = x;
969                        x.parent = n;
970                }
971        }

973        /**
974         * Introduction to Algorithms (CLR) Second Edition
975         * 
976         * <pre>
977         * RB-INSERT-FIXUP(T, z)
978         * while color[p[z]] = RED
979         *    do if p[z] = left[p[p[z]]]
980         *          then y = right[p[p[z]]]
981         *               if color[y] = RED
982         *                  then color[p[z]] = BLACK                                    Case 1
983         *                       color[y] = BLACK                                               Case 1 
984         *                       color[p[p[z]]] = RED                                   Case 1
985         *                       z = p[p[z]]                                                    Case 1
986         *                  else if z = right[p[z]]
987         *                          then z = p[z]                                               Case 2
988         *                               LEFT-ROTATE(T, z)                              Case 2
989         *                       color[p[z]] = BLACK                                    Case 3
990         *                       color[p[p[z]]] = RED                                   Case 3
991         *                       RIGHT-ROTATE(T, p[p[z]])                               Case 3
992         *          else (same as then clause
993         *                        with right and left exchanged)
994         * color[root[T]] = BLACK
995         */
996        private void fixAfterInsertion(Node z) {
997                z.color = RED;
998                while (z.parent.color == RED) {
999                        if (z.parent == z.parent.parent.left) {
1000                                Node y = z.parent.parent.right;
1001                                if (y.color == RED) {
1002                                        z.parent.color = BLACK;
1003                                        y.color = BLACK;
1004                                        z.parent.parent.color = RED;
1005                                        z = z.parent.parent;
1006                                } else {
1007                                        if (z == z.parent.right) {
1008                                                z = z.parent;
1009                                                leftRotate(z);
1010                                        }
1011                                        z.parent.color = BLACK;
1012                                        z.parent.parent.color = RED;
1013                                        rightRotate(z.parent.parent);
1014                                }
1015                        } else {
1016                                Node y = z.parent.parent.left;
1017                                if (y.color == RED) {
1018                                        z.parent.color = BLACK;
1019                                        y.color = BLACK;
1020                                        z.parent.parent.color = RED;
1021                                        z = z.parent.parent;
1022                                } else {
1023                                        if (z == z.parent.left) {
1024                                                z = z.parent;
1025                                                rightRotate(z);
1026                                        }
1027                                        z.parent.color = BLACK;
1028                                        z.parent.parent.color = RED;
1029                                        leftRotate(z.parent.parent);
1030                                }
1031                        }
1032                }
1033                root.color = BLACK;
1034        }

1036        /**
1037         * Introduction to Algorithms (CLR) Second Edition
1038         * 
1039         * <pre>
1040         * RB-DELETE-FIXUP(T, x)
1041         * while x != root[T] and color[x] = BLACK
1042         *    do if x = left[p[x]]
1043         *          then w = right[p[x]]
1044         *               if color[w] = RED
1045         *                  then color[w] = BLACK                                                               Case 1
1046         *                       color[p[x]] = RED                                                              Case 1
1047         *                       LEFT-ROTATE(T, p[x])                                                   Case 1
1048         *                       w = right[p[x]]                                                                Case 1
1049         *               if color[left[w]] = BLACK and color[right[w]] = BLACK
1050         *                  then color[w] = RED                                                                 Case 2
1051         *                       x = p[x]                                                                               Case 2
1052         *                  else if color[right[w]] = BLACK
1053         *                          then color[left[w]] = BLACK                                 Case 3
1054         *                               color[w] = RED                                                 Case 3
1055         *                               RIGHT-ROTATE(T,w)                                              Case 3
1056         *                               w = right[p[x]]                                                Case 3
1057         *                       color[w] = color[p[x]]                                                 Case 4
1058         *                       color[p[x]] = BLACK                                                    Case 4
1059         *                       color[right[w]] = BLACK                                                Case 4
1060         *                       LEFT-ROTATE(T, p[x])                                                   Case 4
1061         *                       x = root[T]                                                                    Case 4
1062         *          else (same as then clause with right and left exchanged)
1063         * color[x] = BLACK
1064         */
1065        private void fixAfterDeletion(Node x) {
1066                while (x != root && x.color == BLACK) {
1067                        if (x == x.parent.left) {
1068                                Node w = x.parent.right;
1069                                if (w.color == RED) {
1070                                        w.color = BLACK;
1071                                        x.parent.color = RED;
1072                                        leftRotate(x.parent);
1073                                        w = x.parent.right;
1074                                }
1075                                if (w.left.color == BLACK && w.right.color == BLACK) {
1076                                        w.color = RED;
1077                                        x = x.parent;
1078                                } else {
1079                                        if (w.right.color == BLACK) {
1080                                                w.left.color = BLACK;
1081                                                w.color = RED;
1082                                                rightRotate(w);
1083                                                w = x.parent.right;
1084                                        }
1085                                        w.color = x.parent.color;
1086                                        x.parent.color = BLACK;
1087                                        x.right.color = BLACK;
1088                                        leftRotate(x.parent);
1089                                        x = root;
1090                                }
1091                        } else {
1092                                Node w = x.parent.left;
1093                                if (w.color == RED) {
1094                                        w.color = BLACK;
1095                                        x.parent.color = RED;
1096                                        rightRotate(x.parent);
1097                                        w = x.parent.left;
1098                                }
1099                                if (w.left.color == BLACK && w.right.color == BLACK) {
1100                                        w.color = RED;
1101                                        x = x.parent;
1102                                } else {
1103                                        if (w.left.color == BLACK) {
1104                                                w.right.color = BLACK;
1105                                                w.color = RED;
1106                                                leftRotate(w);
1107                                                w = x.parent.left;
1108                                        }
1109                                        w.color = x.parent.color;
1110                                        x.parent.color = BLACK;
1111                                        w.left.color = BLACK;
1112                                        rightRotate(x.parent);
1113                                        x = root;
1114                                }
1115                        }
1116                }
1117                x.color = BLACK;
1118        }
