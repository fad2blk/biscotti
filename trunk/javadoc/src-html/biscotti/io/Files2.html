046        /**
047         * Creates any necessary but nonexistent parent directories of the specified
048         * file. Note that if this operation fails it may have succeeded in creating
049         * some (but not all) of the necessary parent directories.
050         * <p>
051         * Unlike {@link Files#createParentDirs(File)} this methods returns the
052         * specified file.
053         * 
054         * @param path
055         *            the specified file
056         * @return the specified file
057         * @throws IOException
058         */
059        public static File createParentDirs(final File path) throws IOException {
060                checkNotNull(path);
061                Files.createParentDirs(path);
062                return path;

065        /**
066         * Returns a {@code File} object representing the canonical path of the
067         * current working directory.
068         * <p>
069         * For example the following code will simulate the Linux/Unix <i>pwd</i>
070         * command: {@code System.out.println(IO.getCurrentDirectory());}

072         * 
073         * @return a {@code File} object representing the current working directory
074         * @throws IOException
075         *             if an I/O error occurs

077        public static File getWorkingDirectory() throws IOException {

081        /**
082         * Returns a print writer to the given file using the specified charset.
083         * 
084         * @param path
085         *            the given file
086         * @param charset
087         *            the character set to use when writing the file
088         * @return a print writer to the given file
089         * @param autoFlush
090         *            if {@code true}, the {@code println}, {@code printf}, and
091         *            {@code format} methods will flush the output buffer
092         * @throws IOException
093         *             if an I/O error occurs
094         * @see Charsets
095         */
096        public static PrintWriter newPrintWriter(final File path,
097                        final Charset charset, final boolean autoFlush) throws IOException {
098                checkNotNull(path);
099                checkNotNull(charset);
100                return new PrintWriter(Files.newWriter(path, charset), autoFlush);
101        }

103        /**
104         * Returns a new {@code PrintStream} which writes to the given file using
105         * the specified charset.
106         * 
107         * @param path
108         *            the specified file
109         * @param charset
110         *            the character set to use when writing to the file
111         * @param autoFlush
112         *            if {@code true}, the output buffer will be flushed whenever a
113         *            byte array is written, one of the println methods is invoked,
114         *            or a newline character or byte ('\n') is written
115         * @return new {@code PrintStream} which writes to the given file
116         * @throws FileNotFoundException
117         *             if an attempt to open the specified file fails
118         */
119        public static PrintStream newPrintStream(final File path,
120                        final Charset charset, final boolean autoFlush)
121                        throws FileNotFoundException {
122                checkNotNull(path);
123                checkNotNull(charset);
124                try {
125                        return new PrintStream(new FileOutputStream(path), autoFlush,
126                                        charset.toString());
127                } catch (UnsupportedEncodingException e) {
128                        throw new Error(); // cannot happen
129                }
130        }

132        /**
133         * Writes a list of strings (separating them using line-termination
134         * characters) to the given file using the specified charset.
135         * 
136         * @param path
137         *            the given file
138         * @param charset
139         *            the character set to use when writing the lines
140         * @param lines
141         *            an iterable of strings to be written to the file
142         * @return the given {@code File} object
143         * @throws IOException
144         *             if an I/O error occurs
145         * @see Charsets
146         */
147        public static File writeLines(final Iterable<String> lines,
148                        final File path, final Charset charset) throws IOException {
149                checkNotNull(path);
150                checkNotNull(charset);
151                checkNotNull(lines);
152                final PrintWriter writer = new PrintWriter(Files
153                                .newOutputStreamSupplier(path).getOutput());
154                for (String line : lines)
155                        writer.println(line);
156                Closeables.closeQuietly(writer);
157                return path;
158        }

160        /**
161         * Returns the specified path string normalized to use the system separator
162         * as specified by {@link File#separator}.
163         * 
164         * @param path
165         *            the specified path
166         * @return the specified path string normalized to use the system separator
167         *         as specified by {@link File#separator}
168         */
169        public static String separatorsToSystem(final String path) {
170                checkNotNull(path);
171                if (File.separator.equals("/"))
172                        return path.replace("\\", "/");
173                else
174                        return path.replace("/", "\\");
175        }

199        /**
200         * Returns all the files and sub-directories in the specified path.
201         * 
202         * @param path
203         *            the specified path
204         * @return all the files and sub-directories in the specified path
205         * @throws IOException
206         *             if an I/O error occurs
207         */
208        public static Iterable<File> walkFileTree(final File path)
209                        throws IOException {
210                checkNotNull(path);
211                checkArgument(path.isDirectory());
212                return walkFileTree(path, FileFilters.TRUE);
213        }

215        /**
216         * Returns all the files and sub-directories in the specified path which
217         * satisfy the given filter.
218         * 
219         * @param path
220         *            the specified path
221         * @param filter
222         *            a file filter
223         * @return all the files and sub-directories in the specified path which
224         *         satisfy the given filter
225         * @throws IOException
226         *             if an I/O error occurs
227         */
228        public static Iterable<File> walkFileTree(final File path,
229                        final FileFilter filter) throws IOException {
230                checkNotNull(path);
231                checkNotNull(filter);
232                final ImmutableSet.Builder<File> builder = ImmutableSet.builder();
233                walkFileTree(builder, path, filter);
234                return builder.build();
235        }

237        /**
238         * Processes and returns all the files and sub-directories in the specified
239         * path which satisfy the given filter.
240         * 
241         * @param path
242         *            the specified directory
243         * @param filter
244         *            a file filter
245         * @param processor
246         *            the {@code FileProcessor} instance used to process the files
247         * @throws IOException
248         *             if an I/O error occurs
249         */
250        public static void walkFileTree(final File path,
251                        final FileFilter filter, final FileProcessor<?> processor)
252                        throws IOException {
253                checkNotNull(path);
254                checkNotNull(filter);
255                checkNotNull(processor);
256                processFileTree(path, filter, processor);
257        }

259        /**
260         * Processes and returns all the files and sub-directories in the specified

262         * 
263         * @param path
264         *            the specified path

266         *            the {@code FileProcessor} instance used to process the files
267         * @throws IOException
268         *             if an I/O error occurs
269         */
270        public static void walkFileTree(final File path,
271                        final FileProcessor<?> processor) throws IOException {
272                checkNotNull(path);
273                checkNotNull(processor);
274                processFileTree(path, FileFilters.TRUE, processor);
275        }
276
277        private static void walkFileTree(final ImmutableSet.Builder<File> builder,
278                        final File path, final FileFilter filter) throws IOException {
279                File[] files = path.listFiles(FileFilters.or(FileFilters.DIRECTORY,
280                                filter));
281                if (files != null)
282                        for (File file : files) {
283                                if (filter.accept(file))
284                                        builder.add(file);
285                                if (file.isDirectory())
286                                        walkFileTree(builder, file, filter);
287                        }
288                else if (filter.accept(path))
289                        builder.add(path);
290        }
291
292        private static void processFileTree(final File path,
293                        final FileFilter filter, final FileProcessor<?> processor)
294                        throws IOException {
295                File[] files = path.listFiles(FileFilters.or(FileFilters.DIRECTORY,
296                                filter));
297                if (files != null)
298                        for (File file : files) {
299                                if (filter.accept(file))
300                                        if (!processor.processFile(file))
301                                                return;
302                                if (file.isDirectory())
303                                        processFileTree(file, filter, processor);
304                        }
305                else if (filter.accept(path))
306                        if (!processor.processFile(path))
307                                return;
308        }

310        /**
311         * Appends a list of strings (separating them using line-termination
312         * characters) to the given file using the specified charset.
313         * 
314         * @param lines
315         *            the character sequence to append
316         * @param to
317         *            the given file
318         * @param charset
319         *            the character set to use when wOriting the lines
320         * @return the given file
321         * @throws IOException
322         *             if an I/O error occurs
323         */
324        public static File appendLines(final List<String> lines, final File to,
325                        final Charset charset) throws IOException {
326                checkNotNull(lines);
327                checkNotNull(to);
328                checkNotNull(charset);
329                Files.append(JOINER.join(lines), to, charset);
330                return to;
331        }
