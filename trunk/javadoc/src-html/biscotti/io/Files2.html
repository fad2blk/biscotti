043        /**
044         * Creates any necessary but nonexistent parent directories of the specified
045         * file. Note that if this operation fails it may have succeeded in creating
046         * some (but not all) of the necessary parent directories.
047         * <p>
048         * Unlike {@link Files#createParentDirs(File)} this methods returns the
049         * specified file.
050         * 
051         * @param path
052         *            the specified file
053         * @return the specified file
054         * @throws IOException
055         */
056        public static File createParentDirs(final File path) throws IOException {
057                checkNotNull(path);
058                Files.createParentDirs(path);

064         * current working directory.
065         * <p>
066         * For example the following code will simulate the Linux/Unix <i>pwd</i>
067         * command: {@code System.out.println(IO.getCurrentDirectory());}

069         * 
070         * @return a {@code File} object representing the current working directory
071         * @throws IOException

075                return new File(".").getCanonicalFile();

094                return new PrintWriter(Files.newWriter(path, charset));

097        /**
098         * Writes a list of strings (separating them using line-termination
099         * characters) to the given file using the specified charset.
100         * 
101         * @param path
102         *            the given file
103         * @param charset
104         *            the character set to use when writing the lines
105         * @param lines
106         *            an iterable of strings to be written to the file
107         * @return the given {@code File} object
108         * @throws IOException
109         *             if an I/O error occurs
110         * @see Charsets
111         */
112        public static File writeLines(final Iterable<String> lines,
113                        final File path, final Charset charset) throws IOException {
114                checkNotNull(path);
115                checkNotNull(charset);
116                checkNotNull(lines);
117                final PrintWriter writer = new PrintWriter(Files
118                                .newOutputStreamSupplier(path).getOutput());
119                for (String line : lines)
120                        writer.println(line);
121                Closeables.closeQuietly(writer);
122                return path;
123        }

125        /**
126         * Returns the specified path string normalized to use the system separator
127         * as specified by {@link File#separator}.
128         * 
129         * @param path
130         *            the specified path
131         * @return the specified path string normalized to use the system separator
132         *         as specified by {@link File#separator}
133         */
134        public static String separatorsToSystem(final String path) {
135                checkNotNull(path);
136                if (File.separator.equals("/"))
137                        return path.replace("\\", "/");
138                else
139                        return path.replace("/", "\\");
140        }

180        /**
181         * Returns all the files and sub-directories in the specified path which

183         * 
184         * @param path

195                checkNotNull(path);
196                checkNotNull(filter);
197                checkArgument(path.isDirectory());
198                return walkFileTree(path, filter, FileProcessors.alwaysTrue());

201        /**
202         * Processes and returns all the files and sub-directories in the specified
203         * path which satisfy the specified filter.
204         * 
205         * @param path
206         *            the specified directory
207         * @param filter
208         *            a file filter
209         * @param processor
210         *            the {@code FileProcessor} instance used to process the files
211         * @return all the files and sub-directories in the specified path which
212         *         satisfy the specified filter
213         * @throws IOException
214         *             if an I/O error occurs
215         */
216        public static Iterable<File> walkFileTree(final File path,
217                        final FileFilter filter, final FileProcessor<?> processor)
218                        throws IOException {
219                checkNotNull(path);
220                checkNotNull(filter);
221                checkNotNull(processor);
222                checkArgument(path.exists());
223                final Builder<File> builder = new ImmutableSet.Builder<File>();
224                walkFileTree(builder, path, filter, processor);
225                return builder.build();
226        }
227
228        /**
229         * Processes and returns all the files and sub-directories in the specified
230         * path.
231         * 
232         * @param path
233         *            the specified path
234         * @param processor
235         *            the {@code FileProcessor} instance used to process the files
236         * @return all the files and sub-directories in the specified path
237         * @throws IOException
238         *             if an I/O error occurs
239         */
240        public static Iterable<File> walkFileTree(final File path,
241                        final FileProcessor<?> processor) throws IOException {
242                checkNotNull(path);
243                checkNotNull(processor);
244                checkArgument(path.isDirectory());
245                return walkFileTree(path, FileFilters.TRUE, processor);
246        }
247
248        private static void walkFileTree(final ImmutableSet.Builder<File> builder,
249                        final File path, final FileFilter filter,
250                        final FileProcessor<?> processor) throws IOException {
251                File[] files = path.listFiles(FileFilters.or(FileFilters.DIRECTORY,
252                                filter));
253                if (files != null)
254                        for (File file : files) {
255                                if (filter.accept(file)) {
256                                        builder.add(file);
257                                        if (!processor.processFile(file))
258                                                return;
259                                }
260                                if (file.isDirectory())
261                                        walkFileTree(builder, file, filter, processor);
262                        }
263                else if (filter.accept(path)) {
264                        builder.add(path);
265                        if (!processor.processFile(path))
266                                return;
267                }
268        }

270        /**
271         * Appends a list of strings (separating them using line-termination
272         * characters) to the given file using the specified charset.
273         * 
274         * @param lines
275         *            the character sequence to append
276         * @param to
277         *            the given file
278         * @param charset
279         *            the character set to use when wOriting the lines
280         * @return the given file
281         * @throws IOException
282         *             if an I/O error occurs
283         */
284        public static File appendLines(final List<String> lines, final File to,
285                        final Charset charset) throws IOException {
286                checkNotNull(lines);
287                checkNotNull(to);
288                checkNotNull(charset);
289                Files.append(JOINER.join(lines), to, charset);
290                return to;
291        }
