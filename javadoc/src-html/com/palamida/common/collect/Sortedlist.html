001package com.palamida.common.collect;
002
003import java.util.ListIterator;
004
005/**
006 * A {@link SortedCollection} which may contain duplicate elements and allows
007 * users to access or remove elements by their integer index.
008 * <p>
009 * This is the <i>list</i> analog to sorted set.
010 * 
011 * @author Zhenya Leonov List
012 * @param <E>
013 *            the type of elements held in this list
014 */
015public interface Sortedlist<E> extends SortedCollection<E> {
016
017        /**
018         * Returns the element at the specified position.
019         * 
020         * @param index
021         *            index of the element to return
022         * @return the element at the specified position
023         * @throws IndexOutOfBoundsException
024         *             if the specified index is out of range
025         */
026        public E get(int index);
027
028        /**
029         * Removes and returns the element at the specified position (optional
030         * operation).
031         * 
032         * @param index
033         *            the index of the element to be removed
034         * @return the element previously at the specified position
035         * @throws IndexOutOfBoundsException
036         *             if the specified index is out of range
037         */
038        public E remove(int index);
039
040        /**
041         * Returns the index of the first occurrence of the specified element, or -1
042         * if the element is not present.
043         * 
044         * @param o
045         *            element to search for
046         * @return the index of the first occurrence of the specified element, or -1
047         *         if the element is not present
048         */
049        public int indexOf(Object o);
050
051        /**
052         * Returns the index of the last occurrence of the specified element, or -1
053         * if the element is not present.
054         * 
055         * @param o
056         *            element to search for
057         * @return the index of the last occurrence of the specified element, or -1
058         *         if the element is not present
059         */
060        public int lastIndexOf(Object o);
061
062        /**
063         * Returns a list iterator over the elements in this sorted-list.
064         * <p>
065         * The returned iterator does not support the
066         * {@link ListIterator#add(Object) add(E)} and
067         * {@link ListIterator#set(Object) set(E)} operations.
068         * 
069         * @return a list iterator over the elements in this sorted-list
070         */
071        public ListIterator<E> listIterator();
072
073        /**
074         * Returns a list iterator over the elements in this sorted-list, starting
075         * at the specified position.
076         * <p>
077         * The returned iterator does not support the
078         * {@link ListIterator#add(Object) add(E)} and
079         * {@link ListIterator#set(Object) set(E)} operations.
080         * 
081         * @param index
082         *            index of the first element to be returned from the collection
083         *            iterator by a call to {@link ListIterator#next() next()}
084         * @return a list iterator over the elements in this sorted-list, starting
085         *         at the specified position
086         */
087        public ListIterator<E> listIterator(int index);
088
089        /**
090         * Returns a view of the portion of this sorted-list whose elements range
091         * between {@code fromIndex}, inclusive, and {@code toIndex}, exclusive. If
092         * {@code fromIndex} and {@code toIndex} are equal, the returned sorted-list
093         * is empty. The returned sorted-list is backed by this sorted-list, so
094         * changes in the returned sorted-list are reflected in this sorted-list,
095         * and vice-versa.
096         * <p>
097         * The returned sorted-list will throw an {@code IllegalArgumentException}
098         * on an attempt to insert an element outside its range.
099         * 
100         * @param fromIndex
101         *            low endpoint (inclusive) of the returned sorted-list
102         * @param toIndex
103         *            high endpoint (exclusive) of the returned sorted-list
104         * @return a view of the portion of this sorted-list within the specified
105         *         range
106         * @throws IndexOutOfBoundsException
107         *             if
108         *             {@code fromIndex < 0 || toIndex > size || fromIndex > toIndex}
109         */
110        public Sortedlist<E> sublist(int fromIndex, int toIndex);
111
112        /**
113         * Returns the hash code value for this sorted-list. The hash code of a
114         * sorted-list is defined to be the result of the following calculation:
115         * 
116         * <pre>
117         * int hashCode = 1;
118         * for (E e : this)
119         *     hashCode = 31 * hashCode + (e == null ? 0 : e.hashCode());
120         * </pre>
121         * 
122         * @return the hash code value for this sorted-list
123         */
124        @Override
125        public int hashCode();
126
127        /**
128         * Compares the specified object with this sorted-list for equality. Returns
129         * {@code true} if and only if the specified object is also a sorted-list,
130         * both sorted-lists have the same size, and all corresponding pairs of
131         * elements in the two sorted-lists are <i>equal</i> according to the
132         * definition of this sorted-list.
133         * 
134         * @return {@code true} if the specified object is {@code equal} to this
135         *         sorted-list
136         */
137        @Override
138        public boolean equals(Object o);
139
140}
