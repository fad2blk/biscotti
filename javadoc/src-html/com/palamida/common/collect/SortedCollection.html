017package com.palamida.common.collect;
018
019import java.util.Collection;
020import java.util.Comparator;
021
022/**
023 * A {@link Collection} which maintains its elements in <i>sorted</i> order.
024 * This interface is the root of all <i>sorted</i> collection interfaces and
025 * implementations.
026 * <p>
027 * Classes which implement this interface (directly or indirectly) are required
028 * to implement the {@code comparator()} method which returns the comparator
029 * used to order the elements in this collection. Essentially this allows
030 * another sorted collection of a desired type to create a copy of this
031 * collection.
032 * <p>
033 * In addition all implementing classes are expected to provide three static
034 * creation methods: {@code create()}, returning a collection that orders its
035 * elements according to their <i>natural ordering</i>,
036 * {@code create(Comparator)} returning a collection which uses the specified
037 * comparator to order its elements, and {@code create(Iterable)} returning a
038 * collection containing the given initial elements. This is simply a refinement
039 * of Java's constructor recommendations, reflecting the new developments of
040 * Java 5.
041 * 
042 * @author Zhenya Leonov
043 * @param <E>
044 *            the type of elements held in this collection
045 */
046public interface SortedCollection<E> extends Collection<E> {
047
048        /**
049         * Returns the comparator used to order the elements in this collection.
050         * Care must be taken when using <i>natural ordering</i>. Implementations
051         * may choose to return a natural order comparator or {@code null}.
052         */
053        public Comparator<? super E> comparator();
054
055}
