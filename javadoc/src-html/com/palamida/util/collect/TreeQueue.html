039/**
040 * An unbounded priority {@link Queue} based on a modified <a
041 * href="http://en.wikipedia.org/wiki/Red-black_tree">Red-Black Tree</a>. The
042 * elements of this queue are sorted according to their <i>natural ordering</i>,
043 * or by an explicit {@link Comparator} provided at creation. Attempting to
044 * remove or insert {@code null} elements is prohibited. Querying for
045 * {@code null} elements is allowed. Inserting non-comparable elements will
046 * result in a {@code ClassCastException}. This queue uses the same general
047 * ordering rules as a {@link PriorityQueue PriorityQueue}. The first element
048 * (the head) of this queue is considered to be the <i>least</i> element with
049 * respect to the specified ordering. A comparator is used (whether or not one
050 * is explicitly provided) to perform all element comparisons. Two elements
051 * which are deemed equal by the comparator's {@code compare(E, E)} method have
052 * equal priority from the standpoint of this queue. Elements with equal
053 * priority are sorted according to their insertion order.
054 * <p>
055 * Besides the regular {@link #peek() peek()}, {@link #poll() poll()},
056 * {@link #remove() remove()} operations specified in the {@code Queue}
057 * interface, this implementation provides additional {@link #peekLast()
058 * peekLast()}, {@link #pollLast() pollLast()}, {@link #removeLast()
059 * removeLast()} methods to examine the elements at the tail of the queue.
060 * <p>
061 * The {@link #iterator() iterator()} and {@link #descendingIterator()} methods
062 * return <i>fail-fast</i> iterators which are guaranteed to traverse the
063 * elements of the queue in ascending and descending priority order,
064 * respectively. Attempts to modify the elements in this queue at any time after
065 * an iterator is created, in any way except through the iterator's own remove
066 * method, will result in a {@code ConcurrentModificationException}.
067 * <p>
068 * This queue is not <i>thread-safe</i>. If multiple threads modify this queue
069 * concurrently it must be synchronized externally.
070 * <p>
071 * The underlying Red-Black Tree provides the following running time compared to
072 * a {@link PriorityQueue PriorityQueue} (where <i>n</i> is the size of this
073 * queue and <i>m</i> is the size of the specified collection which is iterable
074 * in linear time):
075 * <p>
076 * <table border="1" cellpadding="3" cellspacing="1" style="width:400px;">
077 *   <tr>
078 *     <th style="text-align:center;" rowspan="2">Method</th>
079 *     <th style="text-align:center;" colspan="2">Running Time</th>
080 *   </tr>
081 *   <tr>
082 *     <td style="text-align:center;"><b>TreeQueue</b><br>(<i>worst-case</i>)</td>
083 *     <td style="text-align:center;"><b>PriorityQueue</b><br>(<i>worst-case</i>)</td>
084 *   </tr>
085 *   <tr>
086 *     <td>
087 *       {@link #addAll(Collection) addAll(Collection)}<br>
088 *       {@link #containsAll(Collection) containsAll(Collection)}</br>
089 *       {@link #retainAll(Collection) retainAll(Collection)}</br>
090 *       {@link #removeAll(Collection) removeAll(Collection)}
091 *     </td>
092 *     <td colspan="2" style="text-align:center;"><i>O(m log n)</i></td>
093 *   </tr>
094 *   <tr>
095 *     <td>
096 *       {@link #add(Object) add(E)}</br>
097 *       {@link #offer(Object) offer(E)}</br>
098 *       {@link #remove(Object)}
099 *     </td>
100 *     <td colspan="2" style="text-align:center;"><i>O(log n)</i></td>
101 *   </tr>
102 *   <tr>
103 *     <td>
104 *       {@link #contains(Object)}
105 *     </td>
106 *     <td bgcolor="FFCC99" style="text-align:center;"><i>O(log n)</i></td>
107 *     <td bgcolor="FFCCCC" rowspan="2" style="text-align:center;"><i>O(n)</i></td>
108 *   </tr>
109 *   <tr>
110 *     <td>
111 *       {@link #clear()}
112 *     </td>
113 *     <td bgcolor="FFCC99" rowspan="2" style="text-align:center;"><i>O(1)</i></td>
114 *   </tr>
115 *   <tr>
116 *     <td>
117 *       {@link #poll()}</br>
118 *       {@link #remove() remove()}</br>
119 *     </td>
120 *     <td bgcolor="FFCCCC" style="text-align:center;"><i>O(log n)</i></td>
121 *   </tr>
122 *   <tr>
123 *     <td>
124 *       {@link #element() element()}</br>
125 *       {@link #isEmpty() isEmpty()}</br>
126 *       {@link #peek()}</br>
127 *       {@link #size()}
128 *     </td>
129 *     <td colspan="2" style="text-align:center;"><i>O(1)</i></td>
130 *   </tr>
131 * </table>
132 * 
133 * @author Zhenya Leonov
134 * @param <E>
135 *            the type of elements held in this queue
136 * @see TreeBoundedQueue
137 */

154        /**

157         * 
158         * @return a new {@code TreeQueue} that orders its elements according to
159         *         their <i>natural ordering</i>

162                return orderedBy(Ordering.natural()).create();

165        /**
166         * Creates a new {@code TreeQueue} containing the specified initial
167         * elements. If {@code elements} is an instance of {@link SortedSet} ,
168         * {@link PriorityQueue}, {@link MinMaxPriorityQueue}, or
169         * {@code SortedCollection} this queue will be ordered according to the same
170         * ordering. Otherwise, this queue will be ordered according to the
171         * <i>natural ordering</i> of its elements.
172         * 
173         * @param elements
174         *            the collection whose elements are to be placed into the list
175         * @return a new {@code TreeQueue} containing the elements of the specified
176         *         collection
177         * @throws ClassCastException
178         *             if elements of the specified collection cannot be compared to
179         *             one another according to this list's ordering
180         * @throws NullPointerException
181         *             if any of the elements of the specified collection or the
182         *             collection itself is {@code null}
183         */
184        @SuppressWarnings({ "unchecked" })
185        public static <E> TreeQueue<E> from(final Collection<? extends E> elements) {
186                checkNotNull(elements);
187                final Comparator<? super E> comparator;
188                if (elements instanceof SortedSet<?>)
189                        comparator = ((SortedSet<? super E>) elements).comparator();
190                else if (elements instanceof PriorityQueue<?>)
191                        comparator = ((PriorityQueue<? super E>) elements).comparator();
192                else if (elements instanceof SortedCollection<?>)
193                        comparator = ((SortedCollection<? super E>) elements).comparator();
194                else if (elements instanceof MinMaxPriorityQueue<?>)
195                        comparator = ((MinMaxPriorityQueue<? super E>) elements)
196                                        .comparator();
197                else
198                        comparator = (Comparator<? super E>) Ordering.natural();
199                return orderedBy(comparator).create(elements);
200        }

376        /**
377         * Returns an iterator over the elements of this queue in priority order
378         * from first (head) to last (tail).
379         * 
380         * @return an iterator over the elements of this queue in priority order
381         */
382        @Override
383        public Iterator<E> iterator() {
384                return new Iterator<E>() {
385                        private Node next = min;
386                        private Node last = nil;
387                        private int expectedModCount = modCount;
388
389                        @Override
390                        public boolean hasNext() {
391                                return next != nil;
392                        }
393
394                        @Override
395                        public void remove() {
396                                checkForConcurrentModification();
397                                if (last == nil)
398                                        throw new IllegalStateException();
399                                if (last.left != nil && last.right != nil)
400                                        next = last;
401                                delete(last);
402                                expectedModCount = modCount;
403                                last = nil;
404                        }
405
406                        @Override
407                        public E next() {
408                                checkForConcurrentModification();
409                                if (next == nil)
410                                        throw new NoSuchElementException();
411                                last = next;
412                                next = successor(next);
413                                return last.element;
414                        }
415
416                        private void checkForConcurrentModification() {
417                                if (modCount != expectedModCount)
418                                        throw new ConcurrentModificationException();
419                        }
420                };
421        }
422
423        /**
424         * Returns an iterator over the elements of this queue in reverse order from
425         * last (tail) to first (head).
426         * 
427         * @return an iterator over the elements of this queue in reverse order
428         */
429        public Iterator<E> descendingIterator() {
430                return new Iterator<E>() {
431                        private Node next = max;
432                        private Node last = nil;
433                        private int expectedModCount = modCount;
434
435                        @Override
436                        public boolean hasNext() {
437                                return next != nil;
438                        }
439
440                        @Override
441                        public void remove() {
442                                checkForConcurrentModification();
443                                checkState(last != nil);
444                                if (last.left != nil && last.right != nil)
445                                        next = last;
446                                delete(last);
447                                expectedModCount = modCount;
448                                last = nil;
449                        }
450
451                        @Override
452                        public E next() {
453                                checkForConcurrentModification();
454                                if (next == nil)
455                                        throw new NoSuchElementException();
456                                last = next;
457                                next = predecessor(next);
458                                return last.element;
459                        }
460
461                        private void checkForConcurrentModification() {
462                                if (modCount != expectedModCount)
463                                        throw new ConcurrentModificationException();
464                        }
465                };
466        }

515         * Red-Black Tree

518        static enum Color {
519                BLACK, RED;
520        }
521
522        private class Node {
523                private E element = null;
524                private Node parent;
525                private Node left;
526                private Node right;
527                private Color color = BLACK;
528
529                private Node() {
530                        this(null);
531                }
532
533                private Node(final E element) {
534                        this.element = element;
535                        parent = nil;
536                        right = nil;
537                        left = nil;
538                }
539        }

541        /**
542         * Introduction to Algorithms (CLR) Second Edition
543         * 
544         * <pre>
545         * RB-INSERT(T, z)
546         * y = nil[T]
547         * x = root[T]
548         * while x != nil[T]
549         *    do y = x
550         *       if key[z] < key[x]
551         *          then x = left[x]
552         *          else x = right[x]
553         * p[z] = y
554         * if y = nil[T]
555         *    then root[T] = z
556         *    else if key[z] < key[y]
557         *            then left[y] = z
558         *            else right[y] = z
559         * left[z] = nil[T]
560         * right[z] = nil[T]
561         * color[z] = RED
562         * RB-INSERT-FIXUP(T, z)
563         */
564        private void insert(Node z) {
565                size++;
566                modCount++;
567                Node x = root;
568                Node y = nil;
569                while (x != nil) {
570                        y = x;
571                        if (comparator.compare(z.element, x.element) < 0)
572                                x = x.left;
573                        else
574                                x = x.right;
575                }
576                z.parent = y;
577                if (y == nil)
578                        root = z;
579                else if (comparator.compare(z.element, y.element) < 0)
580                        y.left = z;
581                else
582                        y.right = z;
583                fixAfterInsertion(z);
584                if (max == nil || comparator.compare(z.element, max.element) >= 0)
585                        max = z;
586                if (min == nil || comparator.compare(z.element, min.element) < 0)
587                        min = z;
588        }
589
590        private void delete(Node z) {
591                size--;
592                modCount++;
593                Node x, y;
594                if (min == z)
595                        min = successor(z);
596                if (max == z)
597                        max = predecessor(z);
598                if (z.left == nil || z.right == nil)
599                        y = z;
600                else
601                        y = successor(z);
602                if (y.left != nil)
603                        x = y.left;
604                else
605                        x = y.right;
606                x.parent = y.parent;
607                if (y.parent == nil)
608                        root = x;
609                else if (y == y.parent.left)
610                        y.parent.left = x;
611                else
612                        y.parent.right = x;
613                if (y != z)
614                        z.element = y.element;
615                if (y.color == Color.BLACK)
616                        fixAfterDeletion(x);
617        }
618
619        private Node search(final E e) {
620                Node n = root;
621                while (n != nil) {
622                        int cmp = comparator.compare(e, n.element);
623                        if (cmp == 0)
624                                return n;
625                        if (cmp < 0)
626                                n = n.left;
627                        else
628                                n = n.right;
629                }
630                return null;
631        }
632
633        /**
634         * Introduction to Algorithms (CLR) Second Edition
635         * 
636         * <pre>
637         * TREE-SUCCESSOR(x)
638         * if right[x] != NIL
639         *    then return TREE-MINIMUM(right[x])
640         * y = p[x]
641         * while y != NIL and x = right[y]
642         *    do x = y
643         *       y = p[y]
644         * return y
645         */
646        private Node successor(Node x) {
647                if (x == nil)
648                        return nil;
649                if (x.right != nil) {
650                        Node y = x.right;
651                        while (y.left != nil)
652                                y = y.left;
653                        return y;
654                }
655                Node y = x.parent;
656                while (y != nil && x == y.right) {
657                        x = y;
658                        y = y.parent;
659                }
660                return y;
661        }
662
663        private Node predecessor(Node x) {
664                if (x == nil)
665                        return nil;
666                if (x.left != nil) {
667                        Node y = x.left;
668                        while (y.right != nil)
669                                y = y.right;
670                        return y;
671                }
672                Node y = x.parent;
673                while (y != nil && x == y.left) {
674                        x = y;
675                        y = y.left;
676                }
677                return y;
678        }
679
680        /**
681         * Introduction to Algorithms (CLR) Second Edition
682         * 
683         * <pre>
684         * LEFT-ROTATE(T, x)
685         * y = right[x]                                                 Set y.
686         * right[x] = left[y]                                   Turn y's left subtree into x's right subtree.
687         * if left[y] != nil[T]
688         *    then p[left[y]] = x
689         * p[y] = p[x]                                                  Link x's parent to y.
690         * if p[x] = nil[T]
691         *    then root[T] = y
692         *    else if x = left[p[x]]
693         *            then left[p[x]] = y
694         *            else right[p[x]] = y
695         * left[y] = x                                                  Put x on y's left.
696         * p[x] = y
697         */
698        private void leftRotate(final Node x) {
699                if (x != nil) {
700                        Node n = x.right;
701                        x.right = n.left;
702                        if (n.left != nil)
703                                n.left.parent = x;
704                        n.parent = x.parent;
705                        if (x.parent == nil)
706                                root = n;
707                        else if (x.parent.left == x)
708                                x.parent.left = n;
709                        else
710                                x.parent.right = n;
711                        n.left = x;
712                        x.parent = n;
713                }
714        }
715
716        private void rightRotate(final Node x) {
717                if (x != nil) {
718                        Node n = x.left;
719                        x.left = n.right;
720                        if (n.right != nil)
721                                n.right.parent = x;
722                        n.parent = x.parent;
723                        if (x.parent == nil)
724                                root = n;
725                        else if (x.parent.right == x)
726                                x.parent.right = n;
727                        else
728                                x.parent.left = n;
729                        n.right = x;
730                        x.parent = n;
731                }
732        }

734        /**
735         * Introduction to Algorithms (CLR) Second Edition
736         * 
737         * <pre>
738         * RB-INSERT-FIXUP(T, z)
739         * while color[p[z]] = RED
740         *    do if p[z] = left[p[p[z]]]
741         *          then y = right[p[p[z]]]
742         *               if color[y] = RED
743         *                  then color[p[z]] = BLACK                                    Case 1
744         *                       color[y] = BLACK                                               Case 1 
745         *                       color[p[p[z]]] = RED                                   Case 1
746         *                       z = p[p[z]]                                                    Case 1
747         *                  else if z = right[p[z]]
748         *                          then z = p[z]                                               Case 2
749         *                               LEFT-ROTATE(T, z)                              Case 2
750         *                       color[p[z]] = BLACK                                    Case 3
751         *                       color[p[p[z]]] = RED                                   Case 3
752         *                       RIGHT-ROTATE(T, p[p[z]])                               Case 3
753         *          else (same as then clause
754         *                        with right and left exchanged)
755         * color[root[T]] = BLACK
756         */
757        private void fixAfterInsertion(Node z) {
758                z.color = RED;
759                while (z.parent.color == RED) {
760                        if (z.parent == z.parent.parent.left) {
761                                Node y = z.parent.parent.right;
762                                if (y.color == RED) {
763                                        z.parent.color = BLACK;
764                                        y.color = BLACK;
765                                        z.parent.parent.color = RED;
766                                        z = z.parent.parent;
767                                } else {
768                                        if (z == z.parent.right) {
769                                                z = z.parent;
770                                                leftRotate(z);
771                                        }
772                                        z.parent.color = BLACK;
773                                        z.parent.parent.color = RED;
774                                        rightRotate(z.parent.parent);
775                                }
776                        } else {
777                                Node y = z.parent.parent.left;
778                                if (y.color == RED) {
779                                        z.parent.color = BLACK;
780                                        y.color = BLACK;
781                                        z.parent.parent.color = RED;
782                                        z = z.parent.parent;
783                                } else {
784                                        if (z == z.parent.left) {
785                                                z = z.parent;
786                                                rightRotate(z);
787                                        }
788                                        z.parent.color = BLACK;
789                                        z.parent.parent.color = RED;
790                                        leftRotate(z.parent.parent);
791                                }
792                        }
793                }
794                root.color = BLACK;
795        }

797        /**
798         * Introduction to Algorithms (CLR) Second Edition
799         * 
800         * <pre>
801         * RB-DELETE-FIXUP(T, x)
802         * while x != root[T] and color[x] = BLACK
803         *    do if x = left[p[x]]
804         *          then w = right[p[x]]
805         *               if color[w] = RED
806         *                  then color[w] = BLACK                                                               Case 1
807         *                       color[p[x]] = RED                                                              Case 1
808         *                       LEFT-ROTATE(T, p[x])                                                   Case 1
809         *                       w = right[p[x]]                                                                Case 1
810         *               if color[left[w]] = BLACK and color[right[w]] = BLACK
811         *                  then color[w] = RED                                                                 Case 2
812         *                       x = p[x]                                                                               Case 2
813         *                  else if color[right[w]] = BLACK
814         *                          then color[left[w]] = BLACK                                 Case 3
815         *                               color[w] = RED                                                 Case 3
816         *                               RIGHT-ROTATE(T,w)                                              Case 3
817         *                               w = right[p[x]]                                                Case 3
818         *                       color[w] = color[p[x]]                                                 Case 4
819         *                       color[p[x]] = BLACK                                                    Case 4
820         *                       color[right[w]] = BLACK                                                Case 4
821         *                       LEFT-ROTATE(T, p[x])                                                   Case 4
822         *                       x = root[T]                                                                    Case 4
823         *          else (same as then clause with right and left exchanged)
824         * color[x] = BLACK
825         */
826        private void fixAfterDeletion(Node x) {
827                while (x != root && x.color == BLACK) {
828                        if (x == x.parent.left) {
829                                Node w = x.parent.right;
830                                if (w.color == RED) {
831                                        w.color = BLACK;
832                                        x.parent.color = RED;
833                                        leftRotate(x.parent);
834                                        w = x.parent.right;
835                                }
836                                if (w.left.color == BLACK && w.right.color == BLACK) {
837                                        w.color = RED;
838                                        x = x.parent;
839                                } else {
840                                        if (w.right.color == BLACK) {
841                                                w.left.color = BLACK;
842                                                w.color = RED;
843                                                rightRotate(w);
844                                                w = x.parent.right;
845                                        }
846                                        w.color = x.parent.color;
847                                        x.parent.color = BLACK;
848                                        x.right.color = BLACK;
849                                        leftRotate(x.parent);
850                                        x = root;
851                                }
852                        } else {
853                                Node w = x.parent.left;
854                                if (w.color == RED) {
855                                        w.color = BLACK;
856                                        x.parent.color = RED;
857                                        rightRotate(x.parent);
858                                        w = x.parent.left;
859                                }
860                                if (w.left.color == BLACK && w.right.color == BLACK) {
861                                        w.color = RED;
862                                        x = x.parent;
863                                } else {
864                                        if (w.left.color == BLACK) {
865                                                w.right.color = BLACK;
866                                                w.color = RED;
867                                                leftRotate(w);
868                                                w = x.parent.left;
869                                        }
870                                        w.color = x.parent.color;
871                                        x.parent.color = BLACK;
872                                        w.left.color = BLACK;
873                                        rightRotate(x.parent);
874                                        x = root;
875                                }
876                        }
877                }
878                x.color = BLACK;
879        }
