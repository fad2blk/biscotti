044/**
045 * A {@link Sortedlist} implementation based on a modified <a
046 * href="http://en.wikipedia.org/wiki/Skip_list">skip list</a>. Elements are
047 * sorted from <i>least</i> to <i>greatest</i> according to their <i>natural
048 * ordering</i>, or by an explicit {@link Comparator} provided at creation.
049 * Attempting to remove or insert {@code null} elements is prohibited. Querying
050 * for {@code null} elements is allowed. Inserting non-comparable elements will
051 * result in a {@code ClassCastException}.
052 * <p>
053 * The iterators obtained from the {@link #iterator()} and
054 * {@link #listIterator()} methods are <i>fail-fast</i>. Attempts to modify the
055 * elements in this sorted-list at any time after an iterator is created, in any
056 * way except through the iterator's own remove method, will result in a
057 * {@code ConcurrentModificationException}. Further, the list iterator does not
058 * support the {@code add(E)} and {@code set(E)} operations.
059 * <p>
060 * This sorted-list is not <i>thread-safe</i>. If multiple threads modify this
061 * sorted-list concurrently it must be synchronized externally.
062 * <p>
063 * This implementation uses a comparator (whether or not one is explicitly
064 * provided) to perform all element comparisons. Two elements which are deemed
065 * equal by the comparator's {@code compare(E, E)} method are, from the
066 * standpoint of this list, equal. Further, no guarantee is made as to the final
067 * order of <i>equal</i> elements. Ties may be broken arbitrarily.
068 * <p>
069 * The underlying array-based <a
070 * href="http://en.wikipedia.org/wiki/Skip_list">skip list</a> is modified to
071 * provide logarithmic running time for insertion, removal, and <a
072 * href="http://en.wikipedia.org/wiki/Random_access">random access</a> lookup
073 * operations (e.g. get the element at the i<i>th</i> index).
074 * <p>
075 * Invented by <a href="http://www.cs.umd.edu/~pugh/">Bill Pugh</a> in 1990, A
076 * skip list is a probabilistic data structure for maintaining items in sorted
077 * order. Strictly speaking it is impossible to make any hard guarantees
078 * regarding the worst-case performance of this class. Practical performance is
079 * <i>expected</i> to be logarithmic with an extremely high degree of
080 * probability as the list grows.
081 * <p>
082 * The following table summarizes the performance of this class compared to a
083 * {@link Treelist} (where n is the size of this sorted-list and <i>m</i> is the
084 * size of the specified collection which is iterable in linear time):
085 * <p>
086 * <table border="1" cellpadding="3" cellspacing="1" style="width:400px;">
087 *   <tr>
088 *     <th style="text-align:center;" rowspan="2">Method</th>
089 *     <th style="text-align:center;" colspan="2">Running Time</th>
090 *   </tr>
091 *   <tr>
092 *     <td style="text-align:center;"><b>Skiplist</b><br>(<i>expected</i>)</td>
093 *     <td style="text-align:center;"><b>Treelist</b><br>(<i>worst-case</i>)</td>
094 *   </tr>
095 *   <tr>
096 *     <td>
097 *       {@link #addAll(Collection) addAll(Collection)}<br/>
098 *       {@link #containsAll(Collection) containsAll(Collection)}<br/>
099 *       {@link #retainAll(Collection) retainAll(Collection)}<br/>
100 *       {@link #removeAll(Collection) removeAll(Collection)}
101 *     </td>
102 *     <td style="text-align:center;" colspan="2"><i>O(m log n)</i></td>
103 *   </tr>
104 *   <tr>
105 *     <td>
106 *       {@link #indexOf(Object)}<br/>
107 *       {@link #lastIndexOf(Object)}<br/>
108 *       {@link #get(int)}<br/>
109 *       {@link #remove(int)}<br/>
110 *       {@link #listIterator(int)}
111 *     </td>
112 *     <td style="text-align:center;" bgcolor="FFCC99"><i>O(log n)</i></td>
113 *     <td style="text-align:center;" bgcolor="FFCCCC"><i>O(n)</i></td>
114 *   </tr>
115 *   <tr>
116 *     <td>
117 *       {@link #add(Object) add(E)}<br/>
118 *       {@link #contains(Object)}<br/>
119 *       {@link #remove(Object)}
120 *     </td>
121 *     <td style="text-align:center;" colspan="2"><i>O(log n)</i></td>
122 *   </tr>
123 *   <tr>
124 *     <td>
125 *       {@link #clear() clear()}<br/>
126 *       {@link #isEmpty() isEmpty()}<br/>
127 *       {@link #size()}<br/>
128 *     </td>
129 *     <td style="text-align:center;" colspan="2"><i>O(1)</i></td>
130 *   </tr>
131 *   <tr>
132 *     <td>
133 *       {@link Iterator#remove()}<br/>
134 *       {@link ListIterator#remove()}
135 *     </td>
136 *     <td style="text-align:center;" bgcolor="FFCC99"><i>O(log n)</i></td>
137 *     <td style="text-align:center;" bgcolor="FFCCCC"><i>O(1)</i></td>
138 *   </tr>
139 * </table>
140 * <p>
141 * The sub-list views exhibit identical time complexity, with the exception of
142 * the {@code clear()} operation which runs in linear time proportional to the
143 * size of the view.
144 * 
145 * @author Zhenya Leonov
146 * @param <E>
147 *            the type of elements maintained by this list
148 * @see Treelist
149 */
150public class Skiplist<E> extends AbstractCollection<E> implements
151                Sortedlist<E>, Serializable, Cloneable {

155        private static final int MAX_LEVEL = 32;
156        transient int size = 0;

159        private transient Node<E> head = new Node<E>(null, MAX_LEVEL);

166        private Skiplist(final Comparator<? super E> comparator) {
167                this.comparator = comparator;
168                for (int i = 0; i < MAX_LEVEL; i++) {
169                        head.next[i] = head;
170                        head.dist[i] = 1;
171                }
172                head.prev = head;
173        }

186        /**
187         * Creates a new {@code Skiplist} containing the specified initial elements.
188         * If {@code elements} is an instance of {@link SortedSet},
189         * {@link PriorityQueue}, {@link MinMaxPriorityQueue}, or
190         * {@code SortedCollection}, this list will be ordered according to the same
191         * ordering. Otherwise, this list will be ordered according to the
192         * <i>natural ordering</i> of its elements.
193         * 
194         * @param elements
195         *            the collection whose elements are to be placed into the list
196         * @return a new {@code Skiplist} containing the elements of the specified
197         *         collection
198         * @throws ClassCastException
199         *             if elements of the specified collection cannot be compared to
200         *             one another according to this list's ordering
201         * @throws NullPointerException
202         *             if any of the elements of the specified collection or the
203         *             collection itself is {@code null}
204         */
205        @SuppressWarnings({ "unchecked" })
206        public static <E> Skiplist<E> from(final Collection<? extends E> elements) {
207                checkNotNull(elements);
208                final Comparator<? super E> comparator;
209                if (elements instanceof SortedSet<?>)
210                        comparator = ((SortedSet<? super E>) elements).comparator();
211                else if (elements instanceof PriorityQueue<?>)
212                        comparator = ((PriorityQueue<? super E>) elements).comparator();
213                else if (elements instanceof SortedCollection<?>)
214                        comparator = ((SortedCollection<? super E>) elements).comparator();
215                else if (elements instanceof MinMaxPriorityQueue<?>)
216                        comparator = ((MinMaxPriorityQueue<? super E>) elements)
217                                        .comparator();
218                else
219                        comparator = (Comparator<? super E>) Ordering.natural();
220                return orderedBy(comparator).create(elements);
221        }

296        /**
297         * Inserts the specified element into this list in sorted order.
298         */
299        @Override
300        public boolean add(E e) {
301                checkNotNull(e);
302                final int newLevel = randomLevel();
303                Node<E> x = head;
304                Node<E> y = head;
305                int i;
306                int idx = 0;
307                for (i = level - 1; i >= 0; i--) {
308                        while (x.next[i] != y
309                                        && comparator.compare(x.next[i].element, e) < 0) {
310                                idx += x.dist[i];
311                                x = x.next[i];
312                        }
313                        y = x.next[i];
314                        update[i] = x;
315                        index[i] = idx;
316                }
317                if (newLevel > level) {
318                        for (i = level; i < newLevel; i++) {
319                                head.dist[i] = size + 1;
320                                update[i] = head;
321                        }
322                        level = newLevel;
323                }
324                x = new Node<E>(e, newLevel);
325                for (i = 0; i < level; i++) {
326                        if (i > newLevel - 1)
327                                update[i].dist[i]++;
328                        else {
329                                x.next[i] = update[i].next[i];
330                                update[i].next[i] = x;
331                                x.dist[i] = index[i] + update[i].dist[i] - idx;
332                                update[i].dist[i] = idx + 1 - index[i];
333
334                        }
335                }
336                x.prev = update[0];
337                x.next().prev = x;
338                modCount++;
339                size++;
340                return true;
341        }

356        @Override
357        public int indexOf(Object o) {
358                if (o != null) {
359                        Node<E> curr = head;
360                        int idx = 0;
361                        final E element = (E) o;
362                        for (int i = level - 1; i >= 0; i--)
363                                while (curr.next[i] != head
364                                                && comparator.compare(curr.next[i].element, element) < 0) {
365                                        idx += curr.dist[i];
366                                        curr = curr.next[i];
367                                }
368                        curr = curr.next();
369                        if (curr != head && comparator.compare(curr.element, element) == 0)
370                                return idx;
371                }
372                return -1;
373        }

410        @SuppressWarnings("unchecked")
411        @Override
412        public boolean remove(Object o) {
413                checkNotNull(o);
414                final E element = (E) o;
415                Node<E> curr = head;
416                for (int i = level - 1; i >= 0; i--) {
417                        while (curr.next[i] != head
418                                        && comparator.compare(curr.next[i].element, element) < 0)
419                                curr = curr.next[i];
420                        update[i] = curr;
421                }
422                curr = curr.next();
423                if (curr == head || comparator.compare(curr.element, element) != 0)
424                        return false;
425                delete(curr, update);
426                return true;
427        }

429        @Override
430        public E remove(int index) {
431                checkElementIndex(index, size);
432                Node<E> curr = head;
433                int idx = 0;
434                for (int i = level - 1; i >= 0; i--) {
435                        while (idx + curr.dist[i] <= index) {
436                                idx += curr.dist[i];
437                                curr = curr.next[i];
438                        }
439                        update[i] = curr;
440                }
441                curr = curr.next();
442                delete(curr, update);
443                return curr.element;
444        }

453                for (int i = 0; i < MAX_LEVEL; i++) {

470        @Override
471        public boolean equals(Object o) {
472                if (o == this)
473                        return true;
474                if (!(o instanceof Sortedlist))
475                        return false;
476                try {
477                        @SuppressWarnings("unchecked")
478                        final Iterator<E> i = ((Collection<E>) o).iterator();
479                        for (E e : this)
480                                if (comparator.compare(e, i.next()) != 0)
481                                        return false;
482                        return !i.hasNext();
483                } catch (ClassCastException e) {
484                        return false;
485                } catch (NullPointerException e) {
486                        return false;
487                }
488        }

496        /**
497         * Returns a shallow copy of this {@code Skiplist}. The elements themselves
498         * are not cloned.
499         * 
500         * @return a shallow copy of this skip list
501         */
502        @SuppressWarnings("unchecked")
503        @Override
504        public Skiplist<E> clone() throws CloneNotSupportedException {
505                Skiplist<E> clone;
506                try {
507                        clone = (Skiplist<E>) super.clone();
508                } catch (java.lang.CloneNotSupportedException e) {
509                        throw new InternalError();
510                }
511                for (int i = 0; i < MAX_LEVEL; i++) {
512                        clone.head.next[i] = clone.head;
513                        clone.head.dist[i] = 1;
514                }
515                clone.head.prev = clone.head;
516                clone.random = new Random();
517                clone.level = 1;
518                clone.modCount = 0;
519                clone.size = 0;
520                clone.addAll(this);
521                return clone;
522        }

524        private void writeObject(java.io.ObjectOutputStream oos)
525                        throws java.io.IOException {
526                oos.defaultWriteObject();
527                oos.writeInt(size);
528                for (E e : this)
529                        oos.writeObject(e);
530        }

532        @SuppressWarnings("unchecked")
533        private void readObject(java.io.ObjectInputStream ois)
534                        throws java.io.IOException, ClassNotFoundException {
535                ois.defaultReadObject();
536                head = new Node<E>(null, MAX_LEVEL);
537                for (int i = 0; i < MAX_LEVEL; i++) {
538                        head.next[i] = head;
539                        head.dist[i] = 1;
540                }
541                update = new Node[MAX_LEVEL];
542                index = new int[MAX_LEVEL];
543                head.prev = head;
544                random = new Random();
545                level = 1;
546                int size = ois.readInt();
547                for (int i = 0; i < size; i++)
548                        add((E) ois.readObject());
549        }

551        private class ListIteratorImpl implements ListIterator<E> {
552                private Node<E> node;
553                private Node<E> last = null;
554                private final int offset;
555                private int index = 0;
556                private int expectedModCount = modCount;
557
558                private ListIteratorImpl(final int index) {
559                        node = index == 0 ? head.next() : search(index);
560                        offset = index;
561                }
562
563                @Override
564                public void add(E element) {
565                        throw new UnsupportedOperationException();
566                }
567
568                @Override
569                public boolean hasNext() {
570                        return index + offset < size();
571                }
572
573                @Override
574                public boolean hasPrevious() {
575                        return index + offset > 0;
576                }
577
578                @Override
579                public E next() {
580                        checkForConcurrentModification();
581                        if (!hasNext())
582                                throw new NoSuchElementException();
583                        index++;
584                        last = node;
585                        node = node.next();
586                        return last.element;
587                }
588
589                @Override
590                public int nextIndex() {
591                        return index + offset;
592                }
593
594                @Override
595                public E previous() {
596                        checkForConcurrentModification();
597                        if (index == 0)
598                                throw new NoSuchElementException();
599                        index--;
600                        last = node = node.prev;
601                        return node.element;
602                }
603
604                @Override
605                public int previousIndex() {
606                        return index + offset - 1;
607                }
608
609                @Override
610                public void remove() {
611                        checkForConcurrentModification();
612                        checkState(last != null);
613                        Skiplist.this.remove(--index + offset);
614                        expectedModCount = modCount;
615                        last = null;
616                }
617
618                @Override
619                public void set(E element) {
620                        throw new UnsupportedOperationException();
621                }
622
623                void checkForConcurrentModification() {
624                        if (expectedModCount != modCount)
625                                throw new ConcurrentModificationException();
626                }
627        }

631        private static class Node<E> {
632                private E element;
633                private Node<E> prev;
634                private final Node<E>[] next;
635                private final int[] dist;
636
637                @SuppressWarnings("unchecked")
638                private Node(final E element, final int size) {
639                        this.element = element;
640                        next = new Node[size];
641                        dist = new int[size];
642                }
643
644                private Node<E> next() {
645                        return next[0];
646                }
647        }
648
649        private int randomLevel() {
650                int randomLevel = 1;
651                while (randomLevel < MAX_LEVEL - 1 && random.nextDouble() < P)
652                        randomLevel++;
653                return randomLevel;
654        }
655
656        private boolean remove(final Node<E> node) {
657                Node<E> curr = head;
658                for (int i = level - 1; i >= 0; i--) {
659                        while (curr.next[i] != head && curr.next[i] != node)
660                                curr = curr.next[i];
661                        update[i] = curr;
662                }
663                curr = curr.next();
664                delete(curr, update);
665                return true;
666        }
667
668        private void delete(final Node<E> node, final Node<E>[] update) {
669                for (int i = 0; i < level; i++)
670                        if (update[i].next[i] == node) {
671                                update[i].next[i] = node.next[i];
672                                update[i].dist[i] += node.dist[i] - 1;
673                        } else
674                                update[i].dist[i]--;
675                node.next().prev = node.prev;
676                while (head.next[level - 1] == head && level > 1)
677                        level--;
678                modCount++;
679                size--;
680        }
681
682        Node<E> search(final E element) {
683                Node<E> curr = head;
684                for (int i = level - 1; i >= 0; i--)
685                        while (curr.next[i] != head
686                                        && comparator.compare(curr.next[i].element, element) < 0)
687                                curr = curr.next[i];
688                curr = curr.next();
689                if (curr != head && comparator.compare(curr.element, element) == 0)
690                        return curr;
691                return null;
692        }
693
694        private Node<E> search(final int index) {
695                Node<E> curr = head;
696                int idx = -1;
697                for (int i = level - 1; i >= 0; i--)
698                        while (idx + curr.dist[i] <= index) {
699                                idx += curr.dist[i];
700                                curr = curr.next[i];
701                        }
702                return curr;
703        }

706        private final class Sublist extends Skiplist<E> {
707                private final Skiplist<E> list;
708                private int offset;
709                private Node<E> from;
710                private Node<E> to;
711
712                public Sublist(final Skiplist<E> list, final int fromIndex,
713                                final int toIndex) {
714                        super(list.comparator);
715                        this.list = list;
716                        this.modCount = list.modCount;
717                        offset = fromIndex;
718                        this.size = toIndex - fromIndex;
719                        from = list.search(fromIndex);
720                        to = list.search(toIndex - 1);
721                }
722
723                @Override
724                public Skiplist<E> clone() throws CloneNotSupportedException {
725                        throw new CloneNotSupportedException();
726                }

731                        throw new NotSerializableException();

736                                throws IOException, ClassNotFoundException {
737                        throw new NotSerializableException();
738                }

740                private void checkForConcurrentModification() {
741                        if (this.modCount != list.modCount)
742                                throw new ConcurrentModificationException();
743                }
744
745                @Override
746                public int size() {
747                        checkForConcurrentModification();
748                        return this.size;
749                }
750
751                @Override
752                public boolean add(E e) {
753                        checkForConcurrentModification();
754                        checkNotNull(e);
755                        checkArgument(inRange(e, from, to));
756                        list.add(e);
757                        this.modCount = list.modCount;
758                        this.size++;
759                        return true;
760
761                }
762
763                @Override
764                public boolean remove(Object o) {
765                        checkForConcurrentModification();
766                        checkNotNull(o);
767                        @SuppressWarnings("unchecked")
768                        E e = (E) o;
769                        checkArgument(inRange(e, from, to));
770                        if (comparator.compare(e, to.element) == 0) {
771                                list.remove(to);
772                                to = to.prev;
773                        } else
774                                list.remove(e);
775                        this.modCount = list.modCount;
776                        this.size--;
777                        return true;
778                }
779
780                @Override
781                public void clear() {
782                        checkForConcurrentModification();
783                        final Iterator<?> iterator = iterator();
784                        while (iterator.hasNext()) {
785                                iterator.next();
786                                iterator.remove();
787                        }
788                }
789
790                @Override
791                public E get(int index) {
792                        checkForConcurrentModification();
793                        checkElementIndex(index, this.size);
794                        return list.get(index + offset);
795                }
796
797                @Override
798                public E remove(int index) {
799                        checkForConcurrentModification();
800                        checkElementIndex(index, this.size);
801                        if (index == this.size - 1)
802                                to = to.prev;
803                        if (index == 0)
804                                from = from.next();
805                        final E e = list.remove(index + offset);
806                        this.modCount = list.modCount;
807                        this.size--;
808                        return e;
809                }
810
811                @Override
812                public int indexOf(Object o) {
813                        checkForConcurrentModification();
814                        checkNotNull(o);
815                        @SuppressWarnings("unchecked")
816                        E e = (E) o;
817                        if (!inRange(e, from, to))
818                                return -1;
819                        if (comparator.compare(e, from.element) == 0)
820                                return 0;
821                        final int result = list.indexOf(e);
822                        return result == -1 ? -1 : result - offset;
823                }
824
825                @Override
826                public int lastIndexOf(Object o) {
827                        checkForConcurrentModification();
828                        checkNotNull(o);
829                        @SuppressWarnings("unchecked")
830                        E e = (E) o;
831                        if (!inRange(e, from, to))
832                                return -1;
833                        if (comparator.compare(to.element, e) == 0)
834                                return this.size - 1;
835                        final int result = list.lastIndexOf(e);
836                        return result == -1 ? -1 : result - offset;
837                }
838
839                @Override
840                public ListIterator<E> listIterator(final int index) {
841                        checkForConcurrentModification();
842                        checkPositionIndex(index, Sublist.this.size);
843                        return new ListIterator<E>() {
844                                final ListIterator<E> li = list.listIterator(index + offset);
845
846                                @Override
847                                public boolean hasNext() {
848                                        return nextIndex() < Sublist.this.size;
849                                }
850
851                                @Override
852                                public E next() {
853                                        if (hasNext())
854                                                return li.next();
855                                        else
856                                                throw new NoSuchElementException();
857                                }
858
859                                @Override
860                                public boolean hasPrevious() {
861                                        return previousIndex() >= 0;
862                                }
863
864                                public E previous() {
865                                        if (hasPrevious())
866                                                return li.previous();
867                                        else
868                                                throw new NoSuchElementException();
869                                }
870
871                                public int nextIndex() {
872                                        return li.nextIndex() - offset;
873                                }
874
875                                public int previousIndex() {
876                                        return li.previousIndex() - offset;
877                                }
878
879                                @Override
880                                public void remove() {
881                                        li.remove();
882                                        Sublist.this.modCount = list.modCount;
883                                        Sublist.this.size--;
884                                }
885
886                                @Override
887                                public void set(E e) {
888                                        throw new UnsupportedOperationException();
889
890                                }
891
892                                @Override
893                                public void add(E e) {
894                                        throw new UnsupportedOperationException();
895                                }
896
897                        };
898                }

960                @Override
961                Node<E> search(final E e) {
962                        checkForConcurrentModification();
963                        if (!inRange(e, from, to))
964                                return null;
965                        if (comparator.compare(e, from.element) == 0)
966                                return from;
967                        if (comparator.compare(e, to.element) == 0)
968                                return to;
969                        return list.search(e);
970                }
971
972                private boolean inRange(final E e, final Node<E> from, final Node<E> to) {
973                        return (comparator.compare(from.element, e) < 1 && comparator
974                                        .compare(e, to.element) < 1);
975                }
