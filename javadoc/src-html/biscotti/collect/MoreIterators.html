052        /**
053         * Combines multiple list iterators into a single {@code ListIterator}. The
054         * returned list iterator supports the {@code add(E)}, {@code remove()}, and
055         * {@code set(E)} operations if the corresponding backing iterator supports
056         * them.
057         * 
058         * @param inputs
059         *            the source list iterators
060         * @return a {@code ListIterator} consisting of specified list iterators
061         */
062        public static <E> ListIterator<E> concat(final ListIterator<E>... inputs) {
063                checkNotNull(inputs);
064                for (ListIterator<E> li : inputs)
065                        checkNotNull(li);
066                return new ListIterator<E>() {
067                        private int count = 0;
068                        private int index = 0;
069
070                        @Override
071                        public void add(E e) {
072                                inputs[count].add(e);
073                        }
074
075                        @Override
076                        public boolean hasNext() {
077                                while (count < inputs.length) {
078                                        if (inputs[count].hasNext())
079                                                return true;
080                                        count++;
081                                }
082                                count--;
083                                return false;
084                        }
085
086                        @Override
087                        public boolean hasPrevious() {
088                                while (count >= 0) {
089                                        if (inputs[count].hasPrevious())
090                                                return true;
091                                        count--;
092                                }
093                                count = 0;
094                                return false;
095                        }
096
097                        @Override
098                        public E next() {
099                                if (hasNext()) {
100                                        index++;
101                                        return inputs[count].next();
102                                }
103                                throw new NoSuchElementException();
104                        }
105
106                        @Override
107                        public int nextIndex() {
108                                return index;
109                        }
110
111                        @Override
112                        public E previous() {
113                                if (hasPrevious()) {
114                                        index--;
115                                        return inputs[count].previous();
116                                }
117                                throw new NoSuchElementException();
118                        }
119
120                        @Override
121                        public int previousIndex() {
122                                return index - 1;
123                        }
124
125                        @Override
126                        public void remove() {
127                                inputs[count].remove();
128                        }
129
130                        @Override
131                        public void set(E e) {
132                                inputs[count].set(e);
133                        }
134                };
135        }

137        /**
138         * Decorates the given {@code ListIterator} to return only the elements that
139         * are instances of the specified {@code Class}.
140         * 
141         * @param unfiltered
142         *            a list iterator containing objects of any type
143         * @param type
144         *            the type of elements desired
145         * @return an {@code UnmodifiableListIterator} containing all elements of
146         *         the original iterator that were of the requested type

149                        final ListIterator<E> unfiltered, final Class<E> type) {
150                return filter(unfiltered, Predicates.instanceOf(type));

153        /**
154         * Decorates the given {@code ListIterator} to return only the elements that
155         * satisfy the specified {@code Predicate}.
156         * 
157         * @param unfiltered
158         *            the given list iterator
159         * @param predicate
160         *            the specified predicate
161         * @return an {@code UnmodifiableListIterator} which returns only the
162         *         elements of {@code unfiltered} that satisfy {@code predicate}
163         */
164        public static <E> UnmodifiableListIterator<E> filter(
165                        final ListIterator<E> unfiltered,
166                        final Predicate<? super E> predicate) {
167                checkNotNull(unfiltered);
168                checkNotNull(predicate);
169                return new UnmodifiableListIterator<E>() {
170                        public E nextElement;
171                        public E prevElement;
172                        public int index = 0;
173                        public boolean hasNext;
174                        public boolean hasPrev;
175                        public boolean next = false;
176                        public boolean prev = false;
177
178                        @Override
179                        public boolean hasNext() {
180                                if (next && hasNext)
181                                        return true;
182                                prev = false;
183                                while (delegate().hasNext()) {
184                                        nextElement = delegate().next();
185                                        if (predicate.apply(nextElement))
186                                                return next = hasNext = true;
187                                }
188                                return false;
189                        }
190
191                        @Override
192                        public boolean hasPrevious() {
193                                if (prev && hasPrev)
194                                        return true;
195                                next = false;
196                                while (delegate().hasPrevious()) {
197                                        prevElement = delegate().previous();
198                                        if (predicate.apply(prevElement))
199                                                return prev = hasPrev = true;
200                                }
201                                return false;
202                        }
203
204                        @Override
205                        public E next() {
206                                if (!next)
207                                        hasNext();
208                                if (hasNext) {
209                                        index++;
210                                        next = false;
211                                        return nextElement;
212                                } else
213                                        throw new NoSuchElementException();
214                        }
215
216                        @Override
217                        public int nextIndex() {
218                                return index;
219                        }
220
221                        @Override
222                        public E previous() {
223                                if (!prev)
224                                        hasPrevious();
225                                if (hasPrev) {
226                                        index--;
227                                        prev = false;
228                                        return prevElement;
229                                } else
230                                        throw new NoSuchElementException();
231                        }
232
233                        @Override
234                        public int previousIndex() {
235                                return index - 1;
236                        }
237
238                        @Override
239                        protected ListIterator<E> delegate() {
240                                return unfiltered;
241                        }
242                };
243        }

265        /**
266         * Decorates the specified {@code Iterator} to behave like a
267         * {@code ListIterator} by maintaining a {@code List} of all elements
268         * returned from calls to {@code next()}.
269         * <p>
270         * The {@code remove()}, {@code set(E)}, and {@code add(E)} operations are
271         * not supported. Modifying the underlying {@code Iterator} yields
272         * unpredictable results.
273         * 
274         * @param iterator
275         *            the underlying {@code Iterator}
276         * @return a {@code UnmodifiableListIterator} backed by {@code iterator}
277         */
278        public static <E> UnmodifiableListIterator<E> listIterator(
279                        final Iterator<E> iterator) {
280                checkNotNull(iterator);
281                return new UnmodifiableListIterator<E>() {
282                        private final List<E> list = Lists.newArrayList();
283                        private int index = -1;
284
285                        @Override
286                        public boolean hasNext() {
287                                if (index == list.size() - 1)
288                                        return iterator.hasNext();
289                                else
290                                        return true;
291                        }
292
293                        @Override
294                        public E next() {
295                                E nextElement;
296                                if (index == list.size() - 1) {
297                                        nextElement = iterator.next();
298                                        list.add(nextElement);
299                                        index++;
300                                } else
301                                        nextElement = list.get(++index);
302                                return nextElement;
303                        }
304
305                        @Override
306                        public boolean hasPrevious() {
307                                return index >= 0;
308                        }
309
310                        @Override
311                        public E previous() {
312                                if (index < 0)
313                                        throw new NoSuchElementException();
314                                return list.get(index--);
315                        }
316
317                        @Override
318                        public int nextIndex() {
319                                return index + 1;
320                        }
321
322                        @Override
323                        public int previousIndex() {
324                                return index;
325                        }
326
327                        // This method is never used.
328                        @Override
329                        protected ListIterator<E> delegate() {
330                                return this;
331                        }
332                };
333        }

335        /**
336         * Returns an {@code Iterator} over the specified {@code Collection} in
337         * reverse sequential order. If the specified collection is an instance of
338         * {@code Deque} or {@code NavigableSet} the iterator is obtained by
339         * invoking the {@code descendingIterator()} method, else an unmodifiable
340         * iterator is returned by calling the {@code reverseOrder(Iterator)}
341         * method.
342         * 
343         * @param c
344         *            the specified collection
345         * @return an iterator over the specified collection in reverse sequential
346         *         order
347         */
348        public static <E> Iterator<E> reverseOrder(final Collection<? extends E> c) {
349                checkNotNull(c);
350                if (c instanceof Deque<?>)
351                        return ((Deque) c).descendingIterator();
352                if (c instanceof NavigableSet<?>)
353                        return ((NavigableSet) c).descendingIterator();
354                else
355                        return reverseOrder(c.iterator());
356
357        }

359        /**
360         * Returns an {@code UnmodifiableIterator} which reverses the iteration
361         * order of the specified {@code Iterator}. The underlying iterator must be
362         * non-blocking and finite, because it is necessary to iterate through all
363         * its elements on startup.
364         * 
365         * @param iterator
366         *            the underlying {@code Iterator}
367         * @return a reverse view of {@code iterator}
368         */
369        public static <E> UnmodifiableIterator<E> reverseOrder(
370                        final Iterator<? extends E> iterator) {
371                checkNotNull(iterator);
372                final Deque<E> stack = Lists.newLinkedList();
373                while (iterator.hasNext()) {
374                        stack.push(iterator.next());
375                }
376                return new UnmodifiableIterator<E>() {
377                        @Override
378                        public boolean hasNext() {
379                                return !stack.isEmpty();
380                        }
381
382                        @Override
383                        public E next() {
384                                return stack.pop();
385                        }
386                };
387        }

389        /**
390         * Returns a {@code ListIterator} that applies the specified
391         * {@code Function} to each element of the given list iterator.
392         * <p>
393         * The returned list iterator does not support {@code add(E)} and
394         * {@code set(E)} but does support {@code remove()} if the given list
395         * iterator does.
396         * 
397         * @param <E>
398         *            the type of elements traversed by the given list iterator
399         * @param <T>
400         *            the type of output of the specified function
401         * @param fromIterator
402         *            the given list iterator
403         * @param function
404         *            the specified function
405         * @return a list iterator that applies {@code function} to each element of
406         *         {@code fromIterator}
407         */
408        public static <E, T> ListIterator<T> transform(
409                        final ListIterator<E> fromIterator,
410                        final Function<? super E, ? extends T> function) {
411                checkNotNull(fromIterator);
412                checkNotNull(function);
413                return new ForwardingListIterator<T>() {
414                        @Override
415                        public T next() {
416                                return function.apply(fromIterator.next());
417                        }
418
419                        @Override
420                        public void add(T e) {
421                                throw new UnsupportedOperationException();
422                        }
423
424                        @Override
425                        public T previous() {
426                                return function.apply(fromIterator.previous());
427                        }
428
429                        @Override
430                        public void set(T e) {
431                                throw new UnsupportedOperationException();
432                        }
433
434                        @Override
435                        // ok since type T is only used in next, previous, add, and set
436                        protected ListIterator delegate() {
437                                return fromIterator;
438                        }
439                };
440        }
