001package biscotti.base;
002
003import static com.google.common.base.Preconditions.checkNotNull;
004
005import java.util.Arrays;
006
007import com.google.common.base.Function;
008import com.google.common.base.Objects;
009import com.google.common.collect.ObjectArrays;
010
011/**
012 * Static utility methods that operate on or return arrays.
013 * 
014 * @author Zhenya Leonov
015 * @see Arrays
016 * @see ObjectArrays
017 */
018final public class ObjectArrays2 {
019
020        private ObjectArrays2() {
021        };
022
023        /**
024         * Returns a new array containing the result of {@code function.apply(F)} on
025         * each element of the specified array.
026         * 
027         * @param function
028         *            the function to apply on each element of the specified array
029         * @param from
030         *            the array to transform
031         * @return new array containing the result of {@code function.apply(F)} on
032         *         each element of the specified array
033         */
034        @SuppressWarnings("unchecked")
035        public static <F, T> T[] transform(final F[] from,
036                        final Function<? super F, ? extends T> function) {
037                checkNotNull(from);
038                checkNotNull(function);
039                final Object[] to = new Object[from.length];
040                for (int i = 0; i < to.length; i++)
041                        to[i] = function.apply(from[i]);
042                return (T[]) to;
043        }
044
045        /**
046         * Returns {@code true} if the specified array contains the given element,
047         * {@code false} otherwise.
048         * 
049         * @param array
050         *            the specified array
051         * @param element
052         *            the object to look find
053         * @return {@code true} if the specified array contains the given element,
054         *         {@code false} otherwise
055         */
056        public static <T> boolean contains(final T[] array, final Object element) {
057                checkNotNull(array);
058                checkNotNull(element);
059                for (Object o : array)
060                        return Objects.equal(o, element);
061                return false;
062        }
063
064}
